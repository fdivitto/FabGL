120 void GraphicsAdapter::setEmulation(Emulation emulation)
121 {
122  if (m_emulation != emulation) {
123  m_emulation = emulation;
124 
125  m_VGADCtrl.end();
126  freeLUT();
127 
128  switch (m_emulation) {
129 
130  case Emulation::None:
131  //printf("Emulation::None\n");
132  break;
133 
134  case Emulation::PC_Text_40x25_16Colors:
135  //printf("Emulation::PC_Text_40x25_16Colors\n");
136  setFont(&FONT_8x8);
137  setCursorShape(5, 7);
138  m_VGADCtrl.setDrawScanlineCallback(drawScanline_PC_Text_40x25_16Colors, this);
139  m_VGADCtrl.setScanlinesPerCallBack(4);
140  m_VGADCtrl.setResolution(VGA_320x200_70Hz); // VGA_320x200_60HzD as alternative?
141  m_columns = m_VGADCtrl.getViewPortWidth() / m_font.width;
142  m_rows = m_VGADCtrl.getViewPortHeight() / m_font.height;
143  break;
144 
145  case Emulation::PC_Text_80x25_16Colors:
146  //printf("Emulation::PC_Text_80x25_16Colors\n");
147  setFont(&FONT_8x16);
148  setCursorShape(13, 15);
149  m_VGADCtrl.setDrawScanlineCallback(drawScanline_PC_Text_80x25_16Colors, this);
150  m_VGADCtrl.setScanlinesPerCallBack(8);
151  m_VGADCtrl.setResolution(VGA_640x400_70Hz); // VGA_640x400_60Hz as alternative?
152  m_columns = m_VGADCtrl.getViewPortWidth() / m_font.width;
153  m_rows = m_VGADCtrl.getViewPortHeight() / m_font.height;
154  break;
155 
156  case Emulation::PC_Graphics_320x200_4Colors:
157  //printf("Emulation::PC_Graphics_320x200_4Colors\n");
158  m_VGADCtrl.setDrawScanlineCallback(drawScanline_PC_Graphics_320x200_4Colors, this);
159  m_VGADCtrl.setScanlinesPerCallBack(1);
160  m_VGADCtrl.setResolution(VGA_320x200_70Hz); // VGA_320x200_60HzD as alternative?
161  break;
162 
163  case Emulation::PC_Graphics_640x200_2Colors:
164  //printf("Emulation::PC_Graphics_640x200_2Colors\n");
165  m_VGADCtrl.setDrawScanlineCallback(drawScanline_PC_Graphics_640x200_2Colors, this);
166  m_VGADCtrl.setScanlinesPerCallBack(1);
167  m_VGADCtrl.setResolution(VGA_640x200_70Hz); // VGA_640x200_60HzD as alternative?
168  break;
169 
170  case Emulation::PC_Graphics_HGC_720x348:
171  //printf("Emulation::PC_Graphics_HGC_720x348\n");
172  m_VGADCtrl.setDrawScanlineCallback(drawScanline_PC_Graphics_HGC_720x348, this);
173  m_VGADCtrl.setScanlinesPerCallBack(2);
174  m_VGADCtrl.setResolution(VGA_720x348_73Hz); // VGA_720x348_50HzD as alternative?
175  break;
176 
177  }
178 
179  if (m_emulation != Emulation::None) {
180  setupLUT();
181  m_VGADCtrl.run();
182  }
183  }

195 void GraphicsAdapter::setupLUT()
196 {
197  switch (m_emulation) {
198 
199  case Emulation::None:
200  break;
201 
202  case Emulation::PC_Text_80x25_16Colors:
203  case Emulation::PC_Text_40x25_16Colors:
204  // each LUT item contains half pixel (an index to 16 colors palette)
205  if (!m_rawLUT)
206  m_rawLUT = (uint8_t*) heap_caps_malloc(16, MALLOC_CAP_8BIT | MALLOC_CAP_INTERNAL);
207  for (int i = 0; i < 16; ++i)
208  m_rawLUT[i] = m_VGADCtrl.createRawPixel(CGAPalette[i]);
209  break;
210 
211  case Emulation::PC_Graphics_320x200_4Colors:
212  // each LUT item contains four pixels (decodes as four raw bytes)
213  if (!m_rawLUT)
214  m_rawLUT = (uint8_t*) heap_caps_malloc(256 * 4, MALLOC_CAP_8BIT | MALLOC_CAP_INTERNAL);
215  for (int i = 0; i < 256; ++i) {
216  for (int j = 0; j < 4; ++j) {
217  int pixel = (i >> (6 - j * 2)) & 0b11;
218  uint8_t rawPixel = m_VGADCtrl.createRawPixel(pixel == 0 ? CGAPalette[m_PCGraphicsBackgroundColorIndex] : CGAGraphics4ColorsPalette[m_PCGraphicsPaletteInUse][pixel]);
219  m_rawLUT[(i * 4) + (j ^ 2)] = rawPixel;
220  }
221  }
222  break;
223 
224  case Emulation::PC_Graphics_640x200_2Colors:
225  // each LUT item contains eight pixels (decodes as eight raw bytes)
226  if (!m_rawLUT)
227  m_rawLUT = (uint8_t*) heap_caps_malloc(256 * 8, MALLOC_CAP_8BIT | MALLOC_CAP_INTERNAL);
228  for (int i = 0; i < 256; ++i) {
229  for (int j = 0; j < 8; ++j) {
230  bool pixel = (i >> (7 - j)) & 1;
231  uint8_t rawPixel = m_VGADCtrl.createRawPixel(pixel ? CGAPalette[m_PCGraphicsForegroundColorIndex] : RGB222(0, 0, 0));
232  m_rawLUT[(i * 8) + (j ^ 2)] = rawPixel;
233  }
234  }
235  break;
236 
237  case Emulation::PC_Graphics_HGC_720x348:
238  // each LUT item contains eight pixels (decodes as eight raw bytes)
239  if (!m_rawLUT)
240  m_rawLUT = (uint8_t*) heap_caps_malloc(256 * 8, MALLOC_CAP_8BIT | MALLOC_CAP_INTERNAL);
241  for (int i = 0; i < 256; ++i) {
242  for (int j = 0; j < 8; ++j) {
243  bool pixel = (i >> (7 - j)) & 1;
244  uint8_t rawPixel = m_VGADCtrl.createRawPixel(RGB222(pixel * 3, pixel * 3, pixel * 3));
245  m_rawLUT[(i * 8) + (j ^ 2)] = rawPixel;
246  }
247  }
248  break;
249 
250  }
251 }

303 void GraphicsAdapter::setCursorShape(int start, int end)
304 {
305  if (start != m_cursorStart || end != m_cursorEnd) {
306  m_cursorStart = start;
307  m_cursorEnd = end;
308 
309  // readapt start->end to the actual font height to make sure the cursor is always visible
310  if (start <= end && end >= m_font.height) {
311  int h = end - start;
312  end = m_font.height - 1;
313  start = end - h;
314  }
315 
316  int charWidthInBytes = (m_font.width + 7) / 8;
317  if (!m_cursorGlyph)
318  m_cursorGlyph = (uint8_t*) heap_caps_malloc(charWidthInBytes * m_font.height, MALLOC_CAP_8BIT | MALLOC_CAP_INTERNAL);
319  memset(m_cursorGlyph, 0, charWidthInBytes * m_font.height);
320  if (end >= start && start >= 0 && start < m_font.height && end < m_font.height)
321  memset(m_cursorGlyph + (start * charWidthInBytes), 0xff, (end - start + 1) * charWidthInBytes);
322  }
323 }

333 void IRAM_ATTR GraphicsAdapter::drawScanline_PC_Text_40x25_16Colors(void * arg, uint8_t * dest, int scanLine)
334 {
335  auto ga = (GraphicsAdapter*) arg;
336 
337  constexpr int CHARWIDTH = 8;
338  constexpr int CHARHEIGHT = 8;
339  constexpr int CHARWIDTHINBYTES = (CHARWIDTH + 7) / 8;
340  constexpr int CHARSIZEINBYTES = CHARWIDTHINBYTES * CHARHEIGHT;
341  constexpr int COLUMNS = 40;
342  constexpr int SCREENWIDTH = 320;
343  constexpr int LINES = 8;
344 
345  if (scanLine == 0)
346  ++ga->m_frameCounter;
347 
348  int charScanline = scanLine & (CHARHEIGHT - 1);
349  int textRow = scanLine / CHARHEIGHT;
350 
351  uint8_t const * fontData = ga->m_font.data + (charScanline * CHARWIDTHINBYTES);
352 
353  uint8_t * rawLUT = ga->m_rawLUT;
354 
355  uint8_t const * curItem = ga->m_videoBuffer + (textRow * COLUMNS * 2);
356 
357  bool showCursor = ga->m_cursorVisible && ga->m_cursorRow == textRow && ((ga->m_frameCounter & 0x1f) < 0xf);
358  int cursorCol = ga->m_cursorCol;
359 
360  bool bit7blink = ga->m_bit7blink;
361  bool blinktime = bit7blink && !((ga->m_frameCounter & 0x3f) < 0x1f);
362 
363  for (int textCol = 0; textCol < COLUMNS; ++textCol) {
364 
365  int charIdx = *curItem++;
366  int charAttr = *curItem++;
367 
368  bool blink = false;
369  if (bit7blink) {
370  blink = blinktime && (charAttr & 0x80);
371  charAttr &= 0x7f;
372  }
373 
374  uint8_t bg = rawLUT[charAttr >> 4];
375  uint8_t fg = blink ? bg : rawLUT[charAttr & 0xf];
376 
377  const uint8_t colors[2] = { bg, fg };
378 
379  uint8_t const * charBitmapPtr = fontData + charIdx * CHARSIZEINBYTES;
380 
381  auto destptr = dest;
382 
383  if (showCursor && textCol == cursorCol) {
384 
385  uint8_t const * cursorBitmapPtr = ga->m_cursorGlyph + (charScanline * CHARWIDTHINBYTES);
386 
387  for (int charRow = 0; charRow < LINES / 2; ++charRow) {
388 
389  uint32_t charBitmap = *charBitmapPtr | *cursorBitmapPtr;
390 
391  *(destptr + 0) = colors[(bool)(charBitmap & 0x20)];
392  *(destptr + 1) = colors[(bool)(charBitmap & 0x10)];
393  *(destptr + 2) = colors[(bool)(charBitmap & 0x80)];
394  *(destptr + 3) = colors[(bool)(charBitmap & 0x40)];
395  *(destptr + 4) = colors[(bool)(charBitmap & 0x02)];
396  *(destptr + 5) = colors[(bool)(charBitmap & 0x01)];
397  *(destptr + 6) = colors[(bool)(charBitmap & 0x08)];
398  *(destptr + 7) = colors[(bool)(charBitmap & 0x04)];
399 
400  destptr += SCREENWIDTH;
401  charBitmapPtr += CHARWIDTHINBYTES;
402  cursorBitmapPtr += CHARWIDTHINBYTES;
403  }
404 
405  } else {
406 
407  for (int charRow = 0; charRow < LINES / 2; ++charRow) {
408 
409  uint32_t charBitmap = *charBitmapPtr;
410 
411  *(destptr + 0) = colors[(bool)(charBitmap & 0x20)];
412  *(destptr + 1) = colors[(bool)(charBitmap & 0x10)];
413  *(destptr + 2) = colors[(bool)(charBitmap & 0x80)];
414  *(destptr + 3) = colors[(bool)(charBitmap & 0x40)];
415  *(destptr + 4) = colors[(bool)(charBitmap & 0x02)];
416  *(destptr + 5) = colors[(bool)(charBitmap & 0x01)];
417  *(destptr + 6) = colors[(bool)(charBitmap & 0x08)];
418  *(destptr + 7) = colors[(bool)(charBitmap & 0x04)];
419 
420  destptr += SCREENWIDTH;
421  charBitmapPtr += CHARWIDTHINBYTES;
422  }
423 
424  }
425 
426  dest += 8;
427 
428  }
429 }
