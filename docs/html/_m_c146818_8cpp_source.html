30 // MC146818 registers
31 #define REG_SECONDS 0x00 // bin: 0..59, bcd: 00..59
32 #define REG_SECONDS_ALARM 0x01 // like REG_SECONDS or >=0xc0 for don't care
33 #define REG_MINUTES 0x02 // bin: 0..59, bcd: 00..59
34 #define REG_MINUTES_ALARM 0x03 // like REG_MINUTES or >=0xc0 for don't care
35 #define REG_HOURS 0x04 // bin: 1..12 or 0..23, bcd: 01..12 or 00..23 (ORed with 0x80 for PM when range is 1..12)
36 #define REG_HOURS_ALARM 0x05 // like REG_HOURS or >=0xc0 for don't care
37 #define REG_DAYOFWEEK 0x06 // bin: 1..7, bcd: 01..07, (sunday = 1)
38 #define REG_DAYOFMONTH 0x07 // bin: 1..31, bcd: 01..31
39 #define REG_MONTH 0x08 // bin: 1..12, bcd: 01..12
40 #define REG_YEAR 0x09 // bin: 0..99, bcd: 00..99
41 
42 // not MC146818 but anyway filled (to avoid Y2K bug)
43 #define REG_CENTURY 0x32 // bcd: 19 or 20
44 
45 // status and control registers
46 #define REG_A 0x0a
47 #define REG_B 0x0b
48 #define REG_C 0x0c
49 #define REG_D 0x0d
50 
51 // bits of register A
52 #define REGA_RS0 0x01 // R/W, rate selection for square wave gen and Periodic Interrupt
53 #define REGA_RS1 0x02 // R/W
54 #define REGA_RS2 0x04 // R/W
55 #define REGA_RS3 0x08 // R/W
56 #define REGA_DV0 0x10 // R/W, input freq divider
57 #define REGA_DV1 0x20 // R/W
58 #define REGA_DV2 0x40 // R/W
59 #define REGA_UIP 0x80 // R/O, 1 = update in progress
60 
61 // bits of register B
62 #define REGB_DSE 0x01 // R/W, 1 = enabled daylight save
63 #define REGB_H24 0x02 // R/W, 1 = 24h mode, 0 = 12h mode
64 #define REGB_DM 0x04 // R/W, 1 = binary format, 0 = BCD format
65 #define REGB_SQWE 0x08 // R/W, 1 = enable SQWE output
66 #define REGB_UIE 0x10 // R/W, 1 = enable update ended interrupt
67 #define REGB_AIE 0x20 // R/W, 1 = enable alarm interrupt
68 #define REGB_PIE 0x40 // R/W, 1 = enable period interrupts
69 #define REGB_SET 0x80 // R/W, 1 = halt time updates
70 
71 // bits of register C
72 #define REGC_UF 0x10 // R/O, 1 = update ended interrupt flag
73 #define REGC_AF 0x20 // R/O, 1 = alarm interrupt flag
74 #define REGC_PF 0x40 // R/O, 1 = period interrupt flag
75 #define REGC_IRQF 0x80 // R/O, this is "UF & UIE | AF & AIE | PF & PIE"
76 
77 // bits of register D
78 #define REGD_VRT 0x80 // R/O, 1 = valid RAM and time

103 void MC146818::init(char const * NVSNameSpace)
104 {
105  // load registers from NVS
106  if (!m_nvs) {
107  esp_err_t err = nvs_flash_init();
108  if (err == ESP_ERR_NVS_NO_FREE_PAGES || err == ESP_ERR_NVS_NEW_VERSION_FOUND) {
109  nvs_flash_erase();
110  nvs_flash_init();
111  }
112  nvs_open(NVSNameSpace, NVS_READWRITE, &m_nvs);
113  }
114  if (m_nvs) {
115  size_t len = sizeof(m_regs);
116  if (nvs_get_blob(m_nvs, NSVKEY_REGS, m_regs, &len) != ESP_OK) {
117  // first time initialization
118  memset(m_regs, 0, sizeof(m_regs));
119  }
120  }
121 }

124 // reload data from NVS
125 void MC146818::reset()
126 {
127  // set registers bits affected by reset
128  m_regs[REG_B] &= ~(REGB_PIE | REGB_AIE | REGB_UIE | REGB_SQWE);
129  m_regs[REG_C] &= ~(REGC_IRQF | REGC_PF | REGC_AF | REGC_UF);
130  m_regs[REG_D] = REGD_VRT; // power ok
131 
132  m_regSel = 0;
133 }

145 // address:
146 // 1 : register read
147 uint8_t MC146818::read(int address)
148 {
149  uint8_t retval = 0;
150  if (address == 1) {
151  if (m_regSel <= REG_YEAR || m_regSel == REG_CENTURY)
152  updateTime();
153  retval = m_regs[m_regSel];
154  if (m_regSel == REG_C) {
155  // timers are enabled when flags are read
156  enableTimers();
157  // flags are cleared on read (but after retval has been assigned!)
158  m_regs[REG_C] = 0;
159  }
160  //printf("MC146818::read(%02X) => %02X (sel=%02X)\n", address, retval, m_regSel);
161  }
162  return retval;
163 }

166 // address:
167 // 0 : register address port (bits 0-6)
168 // 1 : register write
169 void MC146818::write(int address, uint8_t value)
170 {
171  switch (address) {
172  case 0:
173  m_regSel = value & 0x7f;
174  break;
175  case 1:
176  m_regs[m_regSel] = value;
177  if ( (m_regSel == REG_A && (value & 0xf) != 0) ||
178  (m_regSel == REG_B && (value & (REGB_UIE | REGB_AIE | REGB_PIE)) != 0) ) {
179  // timers are enabled when Rate Selection > 0 or any interrupt is enabled
180  enableTimers();
181  }
182  break;
183  }
184 }

194 // get time from system and fill date/time registers
195 void MC146818::updateTime()
196 {
197  if ((m_regs[REG_B] & REGB_SET) == 0) {
198 
199  time_t now;
200  tm timeinfo;
201 
202  time(&now);
203  localtime_r(&now, &timeinfo);
204 
205  bool binary = m_regs[REG_B] & REGB_DM;
206  bool h24 = m_regs[REG_B] & REGB_H24;
207 
208  int year = (1900 + timeinfo.tm_year); // 1986, 2021, ...
209  int century = year / 100; // 19, 20, ...
210 
211  m_regs[REG_CENTURY] = byteToBCD(century);
212 
213  if (binary) {
214  // binary format
215  m_regs[REG_SECONDS] = imin(timeinfo.tm_sec, 59);
216  m_regs[REG_MINUTES] = timeinfo.tm_min;
217  m_regs[REG_HOURS] = h24 ? timeinfo.tm_hour : (((timeinfo.tm_hour - 1) % 12 + 1) | (timeinfo.tm_hour >= 12 ? 0x80 : 0x00));
218  m_regs[REG_DAYOFWEEK] = timeinfo.tm_wday + 1;
219  m_regs[REG_DAYOFMONTH] = timeinfo.tm_mday;
220  m_regs[REG_MONTH] = timeinfo.tm_mon + 1;
221  m_regs[REG_YEAR] = year - century * 100;
222  } else {
223  // BCD format
224  m_regs[REG_SECONDS] = byteToBCD(imin(timeinfo.tm_sec, 59));
225  m_regs[REG_MINUTES] = byteToBCD(timeinfo.tm_min);
226  m_regs[REG_HOURS] = h24 ? byteToBCD(timeinfo.tm_hour) : (byteToBCD((timeinfo.tm_hour - 1) % 12 + 1) | (timeinfo.tm_hour >= 12 ? 0x80 : 0x00));
227  m_regs[REG_DAYOFWEEK] = byteToBCD(timeinfo.tm_wday + 1);
228  m_regs[REG_DAYOFMONTH] = byteToBCD(timeinfo.tm_mday);
229  m_regs[REG_MONTH] = byteToBCD(timeinfo.tm_mon + 1);
230  m_regs[REG_YEAR] = byteToBCD(year - century * 100);
231  }
232  }
233 }

236 void MC146818::enableTimers()
237 {
238  esp_timer_init(); // can be called multiple times
239 
240  // Setup Periodic Interrupt timer
241  stopPeriodicTimer();
242  int rate = m_regs[REG_A] & 0xf;
243  if (rate > 0) {
244  // supported divider (time base)?
245  int divider = (m_regs[REG_A] >> 4) & 7;
246  if (divider == 2) {
247  // we just support 32768Hz time base
248  static const int RATE2US[16] = { 0, 3906, 7812, 122, 244, 488, 976, 1953, 3906, 7812, 15625, 31250, 62500, 125000, 250000, 500000 };
249  esp_timer_create_args_t args = { 0 };
250  args.callback = periodIntTimerFunc;
251  args.arg = this;
252  args.dispatch_method = ESP_TIMER_TASK;
253  esp_timer_create(&args, &m_periodicIntTimerHandle);
254  esp_timer_start_periodic(m_periodicIntTimerHandle, RATE2US[rate]);
255  //printf("MC146818: Periodic timer started\n");
256  } else {
257  printf("MC146818: Unsupported freq divider %d\n", divider);
258  }
259  }
260 
261  // Setup Alarm and End of Update timer
262  if (!m_endUpdateIntTimerHandle) {
263  esp_timer_create_args_t args = { 0 };
264  args.callback = endUpdateIntTimerFunc;
265  args.arg = this;
266  args.dispatch_method = ESP_TIMER_TASK;
267  esp_timer_create(&args, &m_endUpdateIntTimerHandle);
268  esp_timer_start_periodic(m_endUpdateIntTimerHandle, 1000000); // 1 second
269  //printf("MC146818: Alarm & End of Update timer started\n");
270  }
271 }

294 // Handles Periodic events at specified rate
295 void MC146818::periodIntTimerFunc(void * args)
296 {
297  auto m = (MC146818 *) args;
298 
299  // set periodic flag
300  m->m_regs[REG_C] |= REGC_PF;
301 
302  // trig interrupt?
303  if (m->m_regs[REG_B] & REGB_PIE) {
304  m->m_regs[REG_C] |= REGC_PF | REGC_IRQF;
305  m->m_interruptCallback(m->m_context);
306  }
307 }

310 // Fired every second
311 // Handles Alarm and End Update events
312 void MC146818::endUpdateIntTimerFunc(void * args)
313 {
314  auto m = (MC146818 *) args;
315 
316  if ((m->m_regs[REG_B] & REGB_SET) == 0) {
317 
318  // updating flag
319  m->m_regs[REG_A] |= REGA_UIP;
320 
321  m->updateTime();
322 
323  // alarm?
324  if ( ((m->m_regs[REG_SECONDS_ALARM] & 0xc0) == 0xc0 || (m->m_regs[REG_SECONDS_ALARM] == m->m_regs[REG_SECONDS])) &&
325  ((m->m_regs[REG_MINUTES_ALARM] & 0xc0) == 0xc0 || (m->m_regs[REG_MINUTES_ALARM] == m->m_regs[REG_MINUTES])) &&
326  ((m->m_regs[REG_HOURS_ALARM] & 0xc0) == 0xc0 || (m->m_regs[REG_HOURS_ALARM] == m->m_regs[REG_HOURS])) ) {
327  // yes, set flag
328  m->m_regs[REG_C] |= REGC_AF;
329  }
330 
331  // always signal end update
332  m->m_regs[REG_C] |= REGC_UF;
333 
334  // updating flag
335  m->m_regs[REG_A] &= ~REGA_UIP;
336 
337  // trig interrupt?
338  if ( ((m->m_regs[REG_B] & REGB_UIE) && (m->m_regs[REG_C] & REGC_UF)) ||
339  ((m->m_regs[REG_B] & REGB_AIE) && (m->m_regs[REG_C] & REGC_AF)) ) {
340  // yes
341  m->m_regs[REG_C] |= REGC_IRQF;
342  m->m_interruptCallback(m->m_context);
343  }
344 
345  }
346 }
