54 /*
55  * This is a square wave generator or DMA->GPIO stream generator that uses APLL internal Audio PLL clock.
56  *
57  * When FABGLIB_USE_APLL_AB_COEF = 0 (the default) the frequency range is 2651514 Hz to 62500000 Hz.<br>
58  * Average error is 21 Hz, Minimum error is 0, Maximum error is 1000 Hz except for range 41666667 Hz to 42708333 Hz while
59  * frequency remains fixed at 41666666 Hz (error from 0 Hz to 1041666 Hz) and except for range 42708334 hz to 43748999 Hz while
60  * frequency remains fixed at 43750000 Hz (error from 750001 Hz to 1041666 Hz).
61  *
62  * When FABGLIB_USE_APLL_AB_COEF = 1 the frequency range is 82500 Hz to 62500000 Hz. Unfortunately the output has lot of frequency jittering.<br>
63  * Average error is about 7 Hz, Minimum error is 0, Maximum error is 6349 Hz.
64  */
65 class GPIOStream {
66 
67 public:
68 
69  void begin();
70 
71  /*
72  * Initializes GPIOStream and associate GPIOs to the outputs.
73  *
74  * div1_onGPIO0 If true the undivided frequency is delivered on GPIO0.
75  * div2 Specifies the GPIO where to send frequency / 2 (set GPIO_UNUSED to disable output).
76  * div4 Specifies the GPIO where to send frequency / 4 (set GPIO_UNUSED to disable output).
77  * div8 Specifies the GPIO where to send frequency / 8 (set GPIO_UNUSED to disable output).
78  * div16 Specifies the GPIO where to send frequency / 16 (set GPIO_UNUSED to disable output).
79  * div32 Specifies the GPIO where to send frequency / 32 (set GPIO_UNUSED to disable output).
80  * div64 Specifies the GPIO where to send frequency / 64 (set GPIO_UNUSED to disable output).
81  * div128 Specifies the GPIO where to send frequency / 128 (set GPIO_UNUSED to disable output).
82  * div256 Specifies the GPIO where to send frequency / 256 (set GPIO_UNUSED to disable output).
83  *
84  * Example:
85  *
86  * // Outputs 25Mhz on GPIO0 and 6.25Mhz on GPIO5, for 5 seconds
87  * GPIOStream.begin(true, GPIO_UNUSED, GPIO_NUM_5);
88  * GPIOStream.play(25000000);
89  * delay(5000);
90  * // Outputs 20Mhz on GPIO and 5Mhz on GPIO5, for 10 seconds
91  * GPIOStream.play(20000000);
92  * delay(10000);
93  * GPIOStream.stop();
94  */
95  void begin(bool div1_onGPIO0, gpio_num_t div2 = GPIO_UNUSED, gpio_num_t div4 = GPIO_UNUSED, gpio_num_t div8 = GPIO_UNUSED, gpio_num_t div16 = GPIO_UNUSED, gpio_num_t div32 = GPIO_UNUSED, gpio_num_t div64 = GPIO_UNUSED, gpio_num_t div128 = GPIO_UNUSED, gpio_num_t div256 = GPIO_UNUSED);
96 
97  /*

100  void end();

108  * Example:

112  */
113  void play(int freq, lldesc_t volatile * dmaBuffers = nullptr);
114 
115  /*

117  */
118  void stop();
