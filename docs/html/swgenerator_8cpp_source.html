202 #if FABGLIB_USE_APLL_AB_COEF
203 void floatToFraction(double value, int maxDen, int * num, int * den)
204 {
205  int64_t a, h[3] = { 0, 1, 0 }, k[3] = { 1, 0, 0 };
206  int64_t x, d, n = 1;
207  while (value != floor(value)) {
208  n <<= 1;
209  value *= 2;
210  }
211  d = value;
212  for (int i = 0; i < 64; ++i) {
213  a = n ? d / n : 0;
214  if (i && !a)
215  break;
216  x = d;
217  d = n;
218  n = x % n;
219  x = a;
220  if (k[1] * a + k[0] >= maxDen) {
221  x = (maxDen - k[0]) / k[1];
222  if (x * 2 >= a || k[1] >= maxDen)
223  i = 65;
224  else
225  break;
226  }
227  h[2] = x * h[1] + h[0];
228  h[0] = h[1];
229  h[1] = h[2];
230  k[2] = x * k[1] + k[0];
231  k[0] = k[1];
232  k[1] = k[2];
233  }
234  *den = k[1];
235  *num = h[1];
236 }
237 #endif

349 static void APLLCalcParams(double freq, APLLParams * params, uint8_t * a, uint8_t * b, double * out_freq, double * error)
350 {
351  double FXTAL = FABGLIB_XTAL;
352 
353  *error = 999999999;
354 
355  double apll_freq = freq * 2;
356 
357  for (int o_div = 0; o_div <= 31; ++o_div) {
358 
359  int idivisor = (2 * o_div + 4);
360 
361  for (int sdm2 = 4; sdm2 <= 8; ++sdm2) {
362 
363  // from tables above
364  int minSDM1 = (sdm2 == 4 ? 192 : 0);
365  int maxSDM1 = (sdm2 == 8 ? 128 : 255);
366  // apll_freq = XTAL * (4 + sdm2 + sdm1 / 256) / divisor -> sdm1 = (apll_freq * divisor - XTAL * 4 - XTAL * sdm2) * 256 / XTAL
367  int startSDM1 = ((apll_freq * idivisor - FXTAL * 4.0 - FXTAL * sdm2) * 256.0 / FXTAL);
368 #if FABGLIB_USE_APLL_AB_COEF
369  for (int isdm1 = tmax(minSDM1, startSDM1); isdm1 <= maxSDM1; ++isdm1) {
370 #else
371  int isdm1 = startSDM1; {
372 #endif
373 
374  int sdm1 = isdm1;
375  sdm1 = tmax(minSDM1, sdm1);
376  sdm1 = tmin(maxSDM1, sdm1);
377 
378  // apll_freq = XTAL * (4 + sdm2 + sdm1 / 256 + sdm0 / 65536) / divisor -> sdm0 = (apll_freq * divisor - XTAL * 4 - XTAL * sdm2 - XTAL * sdm1 / 256) * 65536 / XTAL
379  int sdm0 = ((apll_freq * idivisor - FXTAL * 4.0 - FXTAL * sdm2 - FXTAL * sdm1 / 256.0) * 65536.0 / FXTAL);
380  // from tables above
381  sdm0 = (sdm2 == 8 && sdm1 == 128 ? 0 : tmin(255, sdm0));
382  sdm0 = tmax(0, sdm0);
383 
384  // dividend inside 350-500Mhz?
385  double dividend = FXTAL * (4.0 + sdm2 + sdm1 / 256.0 + sdm0 / 65536.0);
386  if (dividend >= 350000000 && dividend <= 500000000) {
387  // adjust output frequency using "b/a"
388  double oapll_freq = dividend / idivisor;
389 
390  // Calculates "b/a", assuming tx_bck_div_num = 1 and clkm_div_num = 2:
391  // freq = apll_clk / (2 + clkm_div_b / clkm_div_a)
392  // abr = clkm_div_b / clkm_div_a
393  // freq = apll_clk / (2 + abr) => abr = apll_clk / freq - 2
394  uint8_t oa = 1, ob = 0;
395 #if FABGLIB_USE_APLL_AB_COEF
396  double abr = oapll_freq / freq - 2.0;
397  if (abr > 0 && abr < 1) {
398  int num, den;
399  floatToFraction(abr, 63, &num, &den);
400  ob = tclamp(num, 0, 63);
401  oa = tclamp(den, 0, 63);
402  }
403 #endif
404 
405  // is this the best?
406  double ofreq = oapll_freq / (2.0 + (double)ob / oa);
407  double err = freq - ofreq;
408  if (abs(err) < abs(*error)) {
409  *params = (APLLParams){(uint8_t)sdm0, (uint8_t)sdm1, (uint8_t)sdm2, (uint8_t)o_div};
410  *a = oa;
411  *b = ob;
412  *out_freq = ofreq;
413  *error = err;
414  if (err == 0.0)
415  return;
416  }
417  }
418  }
419 
420  }
421  }
