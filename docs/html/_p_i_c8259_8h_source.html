32 // PIC 8259 (Programmable Interrupt Controller)
33 
34 // Limitations:
35 // - only 8086 mode
36 // - only single mode
37 // - don't care about input level or edge
38 // - don't care about buffered mode or unbuffered mode
39 // - don't support special fully nested mode
40 // - priority is fixed to IR0 = highest
41 // - don't support Poll command
42 // - don't support Special Mask
43 
44 
45 
46 class PIC8259 {
47 
48 public:
49 
50  void reset();
51 
52  void write(int addr, uint8_t value);
53  uint8_t read(int addr);
54 
55  // Device->8259: a device reports interrupt to 8259
56  bool signalInterrupt(int intnum);
57 
58  // 8259->CPU: 8259 reports interrupt to CPU
59  bool pendingInterrupt() { return m_pendingInterrupt; }
60 
61  // 8259->CPU: 8259 reports interrupt number to CPU
62  int pendingInterruptNum() { return m_baseVector | m_pendingIR; }
63 
64  // CPU->8259: CPU acks the pending interrupt to 8259
65  void ackPendingInterrupt();
66 
67 
68 private:
69 
70  void performEOI();
71  int getHighestPriorityBitNum(uint8_t value);
72  void setPendingInterrupt();
73 
74  uint8_t m_state; // all zeros = ready, bit 0 = waiting for ICW2, bit 1 = waiting for ICW3, bit 2 = waiting for ICW4
75 
76  uint8_t m_baseVector; // bits 3..7 got from ICW2
77  bool m_autoEOI;
78 
79  uint8_t m_IRR; // Interrupt Request Register
80  uint8_t m_ISR; // In-Service Register
81  uint8_t m_IMR; // Interrupt Mask Register
82 
83  // if true port 0 reads ISR, otherwise port 0 reads IRR
84  bool m_readISR;
85 
86  bool m_pendingInterrupt;
87  uint8_t m_pendingIR;
88 
89 };
