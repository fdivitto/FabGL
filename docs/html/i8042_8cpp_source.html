30 // Controller status bits
31 #define STATUS_OBF 0x01 // 0 : Output Buffer Full (0 = output buffer empty)
32 #define STATUS_IBF 0x02 // 1 : Input Buffer Full (0 = input buffer empty)
33 #define STATUS_SYSFLAG 0x04 // 2 : 0 = power on reset, 1 = diagnostic ok
34 #define STATUS_CMD 0x08 // 3 : Command or Data, 0 = write to port 0 (0x60), 1 = write to port 1 (0x64)
35 #define STATUS_INH 0x10 // 4 : Inhibit Switch, 0 = Keyboard inhibited, 1 = Keyboard not inhibited
36 #define STATUS_AOBF 0x20 // 5 : Auxiliary Output Buffer Full, 0 = keyboard data, 1 = mouse data
37 #define STATUS_TIMEOUT 0x40 // 6 : 1 = Timeout Error
38 #define STATUS_PARITY_ERR 0x80 // 7 : 1 = Parity Error

41 // Controller commands
42 #define CTRLCMD_NONE 0x00
43 #define CTRLCMD_GET_COMMAND_BYTE 0x20
44 #define CTRLCMD_READ_CONTROLLER_RAM_BEGIN 0x21
45 #define CTRLCMD_READ_CONTROLLER_RAM_END 0x3f

65 // Command byte bits
66 #define CMDBYTE_ENABLE_KEYBOARD_IRQ 0x01 // 0 : 1 = Keyboard output buffer full causes interrupt (IRQ 1)
67 #define CMDBYTE_ENABLE_MOUSE_IRQ 0x02 // 1 : 1 = Mouse output buffer full causes interrupt (IRQ 12)
68 #define CMDBYTE_SYSFLAG 0x04 // 2 : 1 = System flag after successful controller self-test
69 #define CMDBYTE_UNUSED1 0x08 // 3 : unused (must be 0)
70 #define CMDBYTE_DISABLE_KEYBOARD 0x10 // 4 : 1 = Disable keyboard by forcing the keyboard clock low
71 #define CMDBYTE_DISABLE_MOUSE 0x20 // 5 : 1 = Disable mouse by forcing the mouse serial clock line low
72 #define CMDBYTE_STD_SCAN_CONVERSION 0x40 // 6 : 1 = Standard Scan conversion
73 #define CMDBYTE_UNUSED2 0x80 // 7 : unused (must be 0)

88 void i8042::init()
89 {
90  // because mouse is optional, don't re-try if it is not found (to speed-up boot)
91  Mouse::quickCheckHardware();
92 
93  // keyboard configured on port 0, and optionally mouse on port 1
94  m_PS2Controller.begin(PS2Preset::KeyboardPort0_MousePort1, KbdMode::NoVirtualKeys);
95  m_keyboard = m_PS2Controller.keyboard();
96  m_mouse = m_PS2Controller.mouse();
97 
98  m_STATUS = STATUS_SYSFLAG | STATUS_INH;
99  m_DBBOUT = 0;
100  m_DBBIN = 0;
101  m_commandByte = CMDBYTE_ENABLE_KEYBOARD_IRQ | CMDBYTE_ENABLE_MOUSE_IRQ | CMDBYTE_SYSFLAG | CMDBYTE_STD_SCAN_CONVERSION | CMDBYTE_DISABLE_MOUSE;
102 
103  m_executingCommand = CTRLCMD_NONE;
104  m_writeToMouse = false;
105  m_mousePacketIdx = -1;
106 
107  m_mouseIntTrigs = 0;
108  m_keybIntTrigs = 0;
109 }

112 uint8_t i8042::read(int address)
113 {
114  AutoSemaphore autoSemaphore(m_mutex);
115  switch (address) {
116 
117  // 0 = read 8042 output register (DBBOUT) and set OBF = 0 and AOBF = 0
118  // this is port 0x60 as seen from CPU side
119  case 0:
120  m_STATUS &= ~(STATUS_OBF | STATUS_AOBF);
121  //printf("i8042.read(%02X) => %02X\n", address, m_DBBOUT);
122  return m_DBBOUT;
123 
124  // 1 = read 8042 status register (STATUS)
125  // this is port 0x64 as seen from CPU side
126  case 1:
127  //printf("i8042.read(%02X) => %02X\n", address, m_STATUS);
128  return m_STATUS;
129 
130  default:
131  return 0;
132 
133  }
134 }

137 void i8042::write(int address, uint8_t value)
138 {
139  AutoSemaphore autoSemaphore(m_mutex);
140 
141  switch (address) {
142 
143  // 0 = write 8042 input register (DBBIN), set STATUS_CMD = 0 and STATUS_IBF = 1
144  // this is port 0x60 as seen from CPU side
145  case 0:
146  //printf("i8042.write(%02X, %02X)\n", address, value);
147  m_DBBIN = value;
148  m_STATUS = (m_STATUS & ~STATUS_CMD) | STATUS_IBF;
149  break;
150 
151  // 1 = write 8042 input register (DBBIN), set F1 = 1 and STATUS_IBF = 1
152  // this is port 0x64 as seen from CPU side
153  case 1:
154  //printf("i8042.write(%02X, %02X)\n", address, value);
155  m_DBBIN = value;
156  m_STATUS |= STATUS_CMD | STATUS_IBF;
157  break;
158 
159  }

163 void i8042::tick()
164 {
165  AutoSemaphore autoSemaphore(m_mutex);
166 
167  // something to receive from keyboard?
168  if ((m_STATUS & STATUS_OBF) == 0 && m_keyboard->scancodeAvailable()) {
169  if (m_commandByte & CMDBYTE_STD_SCAN_CONVERSION) {
170  // transform "set 2" scancodes to "set 1"
171  uint8_t scode = Keyboard::convScancodeSet2To1(m_keyboard->getNextScancode()); // "set 1" code (0xf0 doesn't change!)
172  m_DBBOUT = (m_DBBOUT == 0xf0 ? (0x80 | scode) : scode);
173  if (scode != 0xf0) {
174  m_STATUS |= STATUS_OBF;
175  // IR1 (IRQ9) triggered when non break code or when code+break has been received
176  ++m_keybIntTrigs;
177  }
178  } else {
179  // no transform
180  m_DBBOUT = m_keyboard->getNextScancode();
181  m_STATUS |= STATUS_OBF;
182  ++m_keybIntTrigs;
183  }
184  }
185 
186  // something to receive from mouse?
187  if ((m_STATUS & STATUS_OBF) == 0 && (m_mousePacketIdx > -1 || m_mouse->packetAvailable())) {
188  if (m_mousePacketIdx == -1)
189  m_mouse->getNextPacket(&m_mousePacket);
190  m_DBBOUT = m_mousePacket.data[++m_mousePacketIdx];
191  if (m_mousePacketIdx == m_mouse->getPacketSize() - 1)
192  m_mousePacketIdx = -1;
193  m_STATUS |= STATUS_OBF | STATUS_AOBF;
194  ++m_mouseIntTrigs;
195  }
196 
197  // something to execute?
198  if (m_STATUS & STATUS_CMD) {
199  m_STATUS &= ~(STATUS_IBF | STATUS_CMD);
200  execCommand();
201  }
202 
203  // something to execute (with parameters)?
204  if ((m_STATUS & STATUS_IBF) && m_executingCommand != CTRLCMD_NONE) {
205  m_STATUS &= ~STATUS_IBF;
206  execCommand();
207  }
208 
209  // something to send?
210  if (m_STATUS & STATUS_IBF) {
211  m_STATUS &= ~(STATUS_IBF | STATUS_PARITY_ERR);
212  if (m_writeToMouse)
213  m_mouse->sendCommand(m_DBBIN);
214  else
215  m_keyboard->sendCommand(m_DBBIN);
216  m_writeToMouse = false;
217  m_STATUS |= STATUS_PARITY_ERR * m_keyboard->parityError();
218  }
219 
220  // are there interrupts to trig?
221  if (m_keybIntTrigs && trigKeyboardInterrupt())
222  --m_keybIntTrigs;
223  if (m_mouseIntTrigs && trigMouseInterrupt())
224  --m_mouseIntTrigs;
225 }

228 void i8042::execCommand()
229 {
230  uint8_t cmd = m_executingCommand == CTRLCMD_NONE ? m_DBBIN : m_executingCommand;
231 
232  switch (cmd) {
233 
234  case CTRLCMD_GET_COMMAND_BYTE:
235  m_DBBOUT = m_commandByte;
236  m_STATUS |= STATUS_OBF;
237  break;
238 
239  case CTRLCMD_WRITE_COMMAND_BYTE:
240  if (m_executingCommand) {
241  // data received
242  updateCommandByte(m_DBBIN);
243  m_executingCommand = CTRLCMD_NONE;
244  } else {
245  // wait for data
246  m_executingCommand = CTRLCMD_WRITE_COMMAND_BYTE;
247  }
248  break;
249 
250  case CTRLCMD_DISABLE_MOUSE_PORT:
251  enableMouse(false);
252  break;
253 
254  case CTRLCMD_ENABLE_MOUSE_PORT:
255  enableMouse(true);
256  break;
257 
258  case CTRLCMD_TEST_MOUSE_PORT:
259  m_DBBOUT = m_mouse->isMouseAvailable() ? 0x00 : 0x02;
260  m_STATUS |= STATUS_OBF;
261  break;
262 
263  case CTRLCMD_SELF_TEST:
264  m_DBBOUT = 0x55; // no errors!
265  m_STATUS |= STATUS_OBF;
266  break;
267 
268  case CTRLCMD_TEST_KEYBOARD_PORT:
269  m_DBBOUT = m_keyboard->isKeyboardAvailable() ? 0x00 : 0x02;
270  m_STATUS |= STATUS_OBF;
271  break;
272 
273  case CTRLCMD_DISABLE_KEYBOARD:
274  updateCommandByte(m_commandByte | CMDBYTE_DISABLE_KEYBOARD);
275  break;
276 
277  case CTRLCMD_ENABLE_KEYBOARD:
278  updateCommandByte(m_commandByte & ~CMDBYTE_DISABLE_KEYBOARD);
279  break;
280 
281  case CTRLCMD_WRITE_TO_MOUSE:
282  m_writeToMouse = 1;
283  break;
284 
285  case CTRLCMD_SYSTEM_RESET:
286  esp_restart();
287  break;
288 
289  default:
290  printf("8042: unsupported controller command %02X\n", cmd);
291  break;
292  }
293 }

296 void i8042::enableMouse(bool value)
297 {
298  AutoSemaphore autoSemaphore(m_mutex);

303 void i8042::updateCommandByte(uint8_t newValue)
304 {
305  // disable keyboard bit changed?
306  if ((newValue ^ m_commandByte) & CMDBYTE_DISABLE_KEYBOARD) {
307  if (newValue & CMDBYTE_DISABLE_KEYBOARD) {
308  m_keyboard->suspendPort();
309  } else {
310  m_keyboard->resumePort();
311  }
312  }
313 
314  // disable mouse bit changed?
315  if ((newValue ^ m_commandByte) & CMDBYTE_DISABLE_MOUSE) {
316  if (newValue & CMDBYTE_DISABLE_MOUSE) {
317  m_mouse->suspendPort();
318  } else {
319  m_mouse->resumePort();
320  }
321  }
322 
323  m_commandByte = newValue;
324 
325 }
