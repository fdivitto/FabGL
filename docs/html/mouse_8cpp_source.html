87 bool Mouse::reset()
88 {
89  if (s_quickCheckHardware) {
90  m_mouseAvailable = send_cmdReset();
91  } else {
92  // tries up to three times for mouse reset
93  for (int i = 0; i < 3; ++i) {
94  m_mouseAvailable = send_cmdReset();
95  if (m_mouseAvailable)
96  break;
97  vTaskDelay(500 / portTICK_PERIOD_MS);
98  }
99  // give the time to the device to be fully initialized
100  vTaskDelay(200 / portTICK_PERIOD_MS);
101  }
102 
103  // negotiate compatibility and default parameters
104  if (m_mouseAvailable) {
105  // try Intellimouse (three buttons + scroll wheel, 4 bytes packet)
106  if (send_cmdSetSampleRate(200) && send_cmdSetSampleRate(100) && send_cmdSetSampleRate(80) && identify() == PS2DeviceType::MouseWithScrollWheel) {
107  // Intellimouse ok!
108  m_mouseType = Intellimouse;
109  }
110 
111  setSampleRate(60);
112  }
113 
114  return m_mouseAvailable;
115 }

142 // Mouse packet format:
143 // byte 0:
144 // bit 0 = Left Button
145 // bit 1 = Right Button
146 // bit 2 = Middle Button
147 // bit 3 = Always 1
148 // bit 4 = X sign bit
149 // bit 5 = Y sign bit
150 // bit 6 = X overflow
151 // bit 7 = Y overflow
152 // byte 1:
153 // X movement
154 // byte 2:
155 // Y movement
156 // byte 3:
157 // Z movement
158 bool Mouse::decodeMousePacket(MousePacket * mousePacket, MouseDelta * delta)
159 {
160  // the bit 4 of first byte must be always 1
161  if ((mousePacket->data[0] & 8) == 0)
162  return false;
163 
164  m_prevStatus = m_status;
165 
166  // decode packet
167  m_status.buttons.left = (mousePacket->data[0] & 0x01 ? 1 : 0);
168  m_status.buttons.middle = (mousePacket->data[0] & 0x04 ? 1 : 0);
169  m_status.buttons.right = (mousePacket->data[0] & 0x02 ? 1 : 0);
170  if (delta) {
171  delta->deltaX = (int16_t)(mousePacket->data[0] & 0x10 ? 0xFF00 | mousePacket->data[1] : mousePacket->data[1]);
172  delta->deltaY = (int16_t)(mousePacket->data[0] & 0x20 ? 0xFF00 | mousePacket->data[2] : mousePacket->data[2]);
173  delta->deltaZ = (int8_t)(getPacketSize() > 3 ? mousePacket->data[3] : 0);
174  delta->overflowX = (mousePacket->data[0] & 0x40 ? 1 : 0);
175  delta->overflowY = (mousePacket->data[0] & 0x80 ? 1 : 0);
176  delta->buttons = m_status.buttons;
177  }
178 
179  return true;
180 }

232 void Mouse::updateAbsolutePosition(MouseDelta * delta)
233 {
234  const int maxDeltaTimeUS = 500000; // after 0.5s doesn't consider acceleration
235 
236  int dx = delta->deltaX;
237  int dy = delta->deltaY;
238  int dz = delta->deltaZ;
239 
240  int64_t now = esp_timer_get_time();
241  int deltaTime = now - m_prevDeltaTime; // time in microseconds
242 
243  if (deltaTime < maxDeltaTimeUS) {
244 
245  // calculate movement acceleration
246  if (dx != 0 || dy != 0) {
247  int deltaDist = isqrt(dx * dx + dy * dy); // distance in mouse points
248  float vel = (float)deltaDist / deltaTime; // velocity in mousepoints/microsecond
249  float newVel = vel + m_movementAcceleration * vel * vel; // new velocity
250  int newDeltaDist = newVel * deltaTime; // new distance
251  dx = dx * newDeltaDist / deltaDist;
252  dy = dy * newDeltaDist / deltaDist;
253  }
254 
255  // calculate wheel acceleration
256  if (dz != 0) {
257  int deltaDist = abs(dz); // distance in wheel points
258  float vel = (float)deltaDist / deltaTime; // velocity in mousepoints/microsecond
259  float newVel = vel + m_wheelAcceleration * vel * vel; // new velocity
260  int newDeltaDist = newVel * deltaTime; // new distance
261  dz = dz * newDeltaDist / deltaDist;
262  }
263 
264  }
265 
266  m_status.X = tclamp((int)m_status.X + dx, 0, m_area.width - 1);
267  m_status.Y = tclamp((int)m_status.Y - dy, 0, m_area.height - 1);
268  m_status.wheelDelta = dz;
269  m_prevDeltaTime = now;
270 }

273 void Mouse::mouseUpdateTask(void * arg)
274 {
275  constexpr int MAX_TIME_BETWEEN_DATA_US = 500000; // maximum time between data composing a delta frame
276 
277  Mouse * mouse = (Mouse*) arg;
278 
279  int64_t prevDataTime = 0;
280 
281  while (true) {
282 
283  MousePacket mousePacket;
284  int mousePacketLen = 0;
285  while (mousePacketLen < mouse->getPacketSize()) {
286  int r = mouse->getData(-1);
287  if (mouse->parityError() || mouse->syncError()) {
288  mousePacketLen = 0;
289  continue;
290  }
291  int64_t now = esp_timer_get_time();
292  if (mousePacketLen > 0 && prevDataTime > 0 && (now - prevDataTime) > MAX_TIME_BETWEEN_DATA_US) {
293  // too much time elapsed since last byte, start a new delta
294  mousePacketLen = 0;
295  }
296  if (r > -1) {
297  mousePacket.data[mousePacketLen++] = r;
298  prevDataTime = now;
299  }
300  }
301 
302  if (mouse->m_absoluteUpdate) {
303  MouseDelta delta;
304  if (mouse->decodeMousePacket(&mousePacket, &delta)) {
305  mouse->updateAbsolutePosition(&delta);
306 
307  // VGA Controller
308  if (mouse->m_updateDisplayController)
309  mouse->m_updateDisplayController->setMouseCursorPos(mouse->m_status.X, mouse->m_status.Y);
310 
311  // queue (if you need availableStatus() or getNextStatus())
312  if (mouse->m_absoluteQueue) {
313  xQueueSend(mouse->m_absoluteQueue, &mouse->m_status, 0);
314  }
315 
316  if (mouse->m_uiApp) {
317  // generate uiApp events
318  if (mouse->m_prevStatus.X != mouse->m_status.X || mouse->m_prevStatus.Y != mouse->m_status.Y) {
319  // X and Y movement: UIEVT_MOUSEMOVE
320  uiEvent evt = uiEvent(nullptr, UIEVT_MOUSEMOVE);
321  evt.params.mouse.status = mouse->m_status;
322  evt.params.mouse.changedButton = 0;
323  mouse->m_uiApp->postEvent(&evt);
324  }
325  if (mouse->m_status.wheelDelta != 0) {
326  // wheel movement: UIEVT_MOUSEWHEEL
327  uiEvent evt = uiEvent(nullptr, UIEVT_MOUSEWHEEL);
328  evt.params.mouse.status = mouse->m_status;
329  evt.params.mouse.changedButton = 0;
330  mouse->m_uiApp->postEvent(&evt);
331  }
332  if (mouse->m_prevStatus.buttons.left != mouse->m_status.buttons.left) {
333  // left button: UIEVT_MOUSEBUTTONDOWN, UIEVT_MOUSEBUTTONUP
334  uiEvent evt = uiEvent(nullptr, mouse->m_status.buttons.left ? UIEVT_MOUSEBUTTONDOWN : UIEVT_MOUSEBUTTONUP);
335  evt.params.mouse.status = mouse->m_status;
336  evt.params.mouse.changedButton = 1;
337  mouse->m_uiApp->postEvent(&evt);
338  }
339  if (mouse->m_prevStatus.buttons.middle != mouse->m_status.buttons.middle) {
340  // middle button: UIEVT_MOUSEBUTTONDOWN, UIEVT_MOUSEBUTTONUP
341  uiEvent evt = uiEvent(nullptr, mouse->m_status.buttons.middle ? UIEVT_MOUSEBUTTONDOWN : UIEVT_MOUSEBUTTONUP);
342  evt.params.mouse.status = mouse->m_status;
343  evt.params.mouse.changedButton = 2;
344  mouse->m_uiApp->postEvent(&evt);
345  }
346  if (mouse->m_prevStatus.buttons.right != mouse->m_status.buttons.right) {
347  // right button: UIEVT_MOUSEBUTTONDOWN, UIEVT_MOUSEBUTTONUP
348  uiEvent evt = uiEvent(nullptr, mouse->m_status.buttons.right ? UIEVT_MOUSEBUTTONDOWN : UIEVT_MOUSEBUTTONUP);
349  evt.params.mouse.status = mouse->m_status;
350  evt.params.mouse.changedButton = 3;
351  mouse->m_uiApp->postEvent(&evt);
352  }
353  }
354 
355  }
356 
357  } else {
358 
359  xQueueOverwrite(mouse->m_receivedPacket, &mousePacket);
360 
361  }
362 
363  }
364 }
