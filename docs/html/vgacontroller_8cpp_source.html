127 void IRAM_ATTR VGAController::VSyncInterrupt(void * arg)
128 {
129  if (I2S1.int_st.out_eof) {
130  auto VGACtrl = (VGAController*)arg;
131  int64_t startTime = VGACtrl->backgroundPrimitiveTimeoutEnabled() ? esp_timer_get_time() : 0;
132  Rect updateRect = Rect(SHRT_MAX, SHRT_MAX, SHRT_MIN, SHRT_MIN);
133  do {
134  Primitive prim;
135  if (VGACtrl->getPrimitiveISR(&prim) == false)
136  break;
137 
138  VGACtrl->execPrimitive(prim, updateRect, true);
139 
140  if (VGACtrl->m_primitiveProcessingSuspended)
141  break;
142 
143  } while (!VGACtrl->backgroundPrimitiveTimeoutEnabled() || (startTime + VGACtrl->m_maxVSyncISRTime > esp_timer_get_time()));
144  VGACtrl->showSprites(updateRect);
145  }
146  I2S1.int_clr.val = I2S1.int_st.val;
147 }

180 // parameters not checked
181 void IRAM_ATTR VGAController::rawFillRow(int y, int x1, int x2, uint8_t pattern)
182 {
183  auto row = m_viewPort[y];
184  // fill first bytes before full 32 bits word
185  int x = x1;
186  for (; x <= x2 && (x & 3) != 0; ++x) {
187  VGA_PIXELINROW(row, x) = pattern;
188  }
189  // fill whole 32 bits words (don't care about VGA_PIXELINROW adjusted alignment)
190  if (x <= x2) {
191  int sz = (x2 & ~3) - x;
192  memset((void*)(row + x), pattern, sz);
193  x += sz;
194  }
195  // fill last unaligned bytes
196  for (; x <= x2; ++x) {
197  VGA_PIXELINROW(row, x) = pattern;
198  }
199 }

213 // swaps all pixels inside the range x1...x2 of yA and yB
214 // parameters not checked
215 void IRAM_ATTR VGAController::swapRows(int yA, int yB, int x1, int x2)
216 {
217  uint8_t * rowA = (uint8_t*) m_viewPort[yA];
218  uint8_t * rowB = (uint8_t*) m_viewPort[yB];
219  // swap first bytes before full 32 bits word
220  int x = x1;
221  for (; x <= x2 && (x & 3) != 0; ++x)
222  tswap(VGA_PIXELINROW(rowA, x), VGA_PIXELINROW(rowB, x));
223  // swap whole 32 bits words (don't care about VGA_PIXELINROW adjusted alignment)
224  uint32_t * a = (uint32_t*)(rowA + x);
225  uint32_t * b = (uint32_t*)(rowB + x);
226  for (int right = (x2 & ~3); x < right; x += 4)
227  tswap(*a++, *b++);
228  // swap last unaligned bytes
229  for (x = (x2 & ~3); x <= x2; ++x)
230  tswap(VGA_PIXELINROW(rowA, x), VGA_PIXELINROW(rowB, x));
231 }

252 // scroll < 0 -> scroll UP
253 // scroll > 0 -> scroll DOWN
254 // Speciying horizontal scrolling region slow-down scrolling!
255 void IRAM_ATTR VGAController::VScroll(int scroll, Rect & updateRect)
256 {
257  genericVScroll(scroll, updateRect,
258  [&] (int yA, int yB, int x1, int x2) { swapRows(yA, yB, x1, x2); }, // swapRowsCopying
259  [&] (int yA, int yB) { tswap(m_viewPort[yA], m_viewPort[yB]); }, // swapRowsPointers
260  [&] (int y, int x1, int x2, RGB888 pattern) { rawFillRow(y, x1, x2, pattern); } // rawFillRow
261  );
262 
263  if (scroll != 0) {
264  // reassign DMA pointers
265  int viewPortBuffersPerLine = 0;
266  int linePos = 1;
267  switch (m_timings.HStartingBlock) {
268  case VGAScanStart::FrontPorch:
269  // FRONTPORCH -> SYNC -> BACKPORCH -> VISIBLEAREA
270  viewPortBuffersPerLine = (m_viewPortCol + m_viewPortWidth) < m_timings.HVisibleArea ? 3 : 2;
271  break;
272  case VGAScanStart::Sync:
273  // SYNC -> BACKPORCH -> VISIBLEAREA -> FRONTPORCH
274  viewPortBuffersPerLine = 3;
275  break;
276  case VGAScanStart::BackPorch:
277  // BACKPORCH -> VISIBLEAREA -> FRONTPORCH -> SYNC
278  viewPortBuffersPerLine = 3;
279  break;
280  case VGAScanStart::VisibleArea:
281  // VISIBLEAREA -> FRONTPORCH -> SYNC -> BACKPORCH
282  viewPortBuffersPerLine = m_viewPortCol > 0 ? 3 : 2;
283  linePos = m_viewPortCol > 0 ? 1 : 0;
284  break;
285  }
286  const int Y1 = paintState().scrollingRegion.Y1;
287  const int Y2 = paintState().scrollingRegion.Y2;
288  for (int i = Y1, idx = Y1 * m_timings.scanCount; i <= Y2; ++i)
289  for (int scan = 0; scan < m_timings.scanCount; ++scan, ++idx)
290  setDMABufferView(m_viewPortRow * m_timings.scanCount + idx * viewPortBuffersPerLine + linePos, i, scan, m_viewPort, false);
291  }
292 }

295 // Scrolling by 1, 2, 3 and 4 pixels is optimized. Also scrolling multiples of 4 (8, 16, 24...) is optimized.
296 // Scrolling by other values requires up to three steps (scopose scrolling by 1, 2, 3 or 4): for example scrolling by 5 is scomposed to 4 and 1, scrolling
297 // by 6 is 4 + 2, etc.
298 // Horizontal scrolling region start and size (X2-X1+1) must be aligned to 32 bits, otherwise the unoptimized (very slow) version is used.
299 void IRAM_ATTR VGAController::HScroll(int scroll, Rect & updateRect)
300 {
301  hideSprites(updateRect);
302  uint8_t pattern8 = preparePixel(getActualBrushColor());
303  uint16_t pattern16 = pattern8 << 8 | pattern8;
304  uint32_t pattern32 = pattern16 << 16 | pattern16;
305 
306  int Y1 = paintState().scrollingRegion.Y1;
307  int Y2 = paintState().scrollingRegion.Y2;
308  int X1 = paintState().scrollingRegion.X1;
309  int X2 = paintState().scrollingRegion.X2;
310 
311  int width = X2 - X1 + 1;
312  int width32 = width >> 2;
313  bool HScrolllingRegionAligned = ((X1 & 3) == 0 && (width & 3) == 0);
314 
315  if (scroll < 0) {
316  // scroll left
317  for (int y = Y1; y <= Y2; ++y) {
318  if (HScrolllingRegionAligned) {
319  // aligned horizontal scrolling region, fast version
320  uint8_t * row = (uint8_t*) (m_viewPort[y] + X1);
321  for (int s = -scroll; s > 0;) {
322  if (s >= 4) {
323  // scroll left 4, 8 12, etc.. pixels moving 32 bit words
324  uint8_t * w = row;
325  int sz = (s & ~3) >> 2;
326  for (int i = 0; i < width32 - sz; ++i, w += 4)
327  *((uint32_t*)w) = *((uint32_t*)w + sz);
328  for (int i = tmax(0, width32 - sz); i < width32; ++i, w += 4)
329  *((uint32_t*)w) = pattern32;
330  s -= (s & ~3);
331  } else if ((s & 3) == 3) {
332  // scroll left 3 pixels swapping 8 bit words
333  uint8_t * b = row;
334  for (int i = 1; i < width32; ++i, b += 4) {
335  b[2] = b[1];
336  b[1] = b[4];
337  b[0] = b[7];
338  b[3] = b[6];
339  }
340  b[2] = b[1];
341  b[1] = b[0] = b[3] = pattern8;
342  s -= 3;
343  } else if (s & 2) {
344  // scroll left 2 pixels swapping 16 bit words
345  uint16_t * w = (uint16_t*) row;
346  for (int i = 1; i < width32; ++i, w += 2) {
347  w[1] = w[0];
348  w[0] = w[3];
349  }
350  w[1] = w[0];
351  w[0] = pattern16;
352  s -= 2;
353  } else if (s & 1) {
354  // scroll left 1 pixel by rotating 32 bit words
355  uint8_t * w = row;
356  for (int i = 1; i < width32; ++i, w += 4) {
357  *((uint32_t*)w) = *((uint32_t*)w) >> 8 | *((uint32_t*)w) << 24;
358  w[1] = w[6];
359  }
360  *((uint32_t*)w) = *((uint32_t*)w) >> 8 | *((uint32_t*)w) << 24;
361  w[1] = pattern8;
362  --s;
363  }
364  }
365  } else {
366  // unaligned horizontal scrolling region, fallback to slow version
367  uint8_t * row = (uint8_t*) m_viewPort[y];
368  for (int x = X1; x <= X2 + scroll; ++x)
369  VGA_PIXELINROW(row, x) = VGA_PIXELINROW(row, x - scroll);
370  // fill right area with brush color
371  for (int x = X2 + 1 + scroll; x <= X2; ++x)
372  VGA_PIXELINROW(row, x) = pattern8;
373  }
374  }
375  } else if (scroll > 0) {
376  // scroll right
377  for (int y = Y1; y <= Y2; ++y) {
378  if (HScrolllingRegionAligned) {
379  // aligned horizontal scrolling region, fast version
380  uint8_t * row = (uint8_t*) (m_viewPort[y] + X1);
381  for (int s = scroll; s > 0;) {
382  if (s >= 4) {
383  // scroll right 4, 8 12, etc.. pixels moving 32 bit words
384  int sz = (s & ~3) >> 2;
385  uint8_t * w = row + width - 4;
386  for (int i = 0; i < width32 - sz; ++i, w -= 4)
387  *((uint32_t*)w) = *((uint32_t*)w - sz);
388  for (int i = tmax(0, width32 - sz); i < width32; ++i, w -= 4)
389  *((uint32_t*)w) = pattern32;
390  s -= (s & ~3);
391  } else if ((s & 3) == 3) {
392  // scroll right 3 pixels swapping 8 bit words
393  uint8_t * b = row + width - 4;
394  for (int i = 1; i < width32; ++i, b -= 4) {
395  b[0] = b[-3];
396  b[1] = b[2];
397  b[2] = b[-1];
398  b[3] = b[-4];
399  }
400  b[1] = b[2];
401  b[0] = b[2] = b[3] = pattern8;
402  s -= 3;
403  } else if (s & 2) {
404  // scroll right 2 pixels swapping 16 bit words
405  uint16_t * w = (uint16_t*) (row + width - 4);
406  for (int i = 1; i < width32; ++i, w -= 2) {
407  w[0] = w[1];
408  w[1] = w[-2];
409  }
410  w[0] = w[1];
411  w[1] = pattern16;
412  s -= 2;
413  } else if (s & 1) {
414  // scroll right 1 pixel by rotating 32 bit words
415  uint8_t * w = row + width - 4;
416  for (int i = 1; i < width32; ++i, w -= 4) {
417  *((uint32_t*)w) = *((uint32_t*)w) << 8 | *((uint32_t*)w) >> 24;
418  w[2] = w[-3];
419  }
420  *((uint32_t*)w) = *((uint32_t*)w) << 8 | *((uint32_t*)w) >> 24;
421  w[2] = pattern8;
422  --s;
423  }
424  }
425  } else {
426  // unaligned horizontal scrolling region, fallback to slow version
427  uint8_t * row = (uint8_t*) m_viewPort[y];
428  for (int x = X2 - scroll; x >= X1; --x)
429  VGA_PIXELINROW(row, x + scroll) = VGA_PIXELINROW(row, x);
430  // fill left area with brush color
431  for (int x = X1; x < X1 + scroll; ++x)
432  VGA_PIXELINROW(row, x) = pattern8;
433  }
434  }
435 
436  }
437 }
