64 bool I2C::begin(gpio_num_t SDAGPIO, gpio_num_t SCLGPIO)
65 {
66  m_SDAGPIO = SDAGPIO;
67  m_SCLGPIO = SCLGPIO;
68 
69  m_eventGroup = xEventGroupCreate();
70 
71  // why a task? Because esp32 i2c communications must be done on the same core
72  // must be pinned to one core (0 in this case)
73  xTaskCreatePinnedToCore(&commTaskFunc, "", I2C_COMMTASK_STACK, this, I2C_COMMTASK_PRIORITY, &m_commTaskHandle, 0);
74 
75  // wait for commTaskFunc() ends initialization
76  xEventGroupWaitBits(m_eventGroup, EVTGROUP_DONE, true, false, portMAX_DELAY);
77 
78  // ready to accept jobs
79  xEventGroupSetBits(m_eventGroup, EVTGROUP_READY);
80 
81  return m_i2c != nullptr;
82 }

97  m_eventGroup = nullptr;

101 bool I2C::write(int address, uint8_t * buffer, int size, int frequency, int timeOutMS)
102 {
103  // wait for I2C to be ready
104  xEventGroupWaitBits(m_eventGroup, EVTGROUP_READY, true, false, portMAX_DELAY);
105 
106  m_jobInfo.frequency = frequency;
107  m_jobInfo.address = address;
108  m_jobInfo.buffer = buffer;
109  m_jobInfo.size = size;
110  m_jobInfo.timeout = timeOutMS;
111 
112  // unlock comm task for writing
113  // wait for comm task to finish job
114  xEventGroupSync(m_eventGroup, EVTGROUP_WRITE, EVTGROUP_DONE, portMAX_DELAY);
115 
116 
117  bool ret = (m_jobInfo.lastError == I2C_ERROR_OK);
118 
119  // makes I2C ready for new requests
120  xEventGroupSetBits(m_eventGroup, EVTGROUP_READY);
121 
122  return ret;
123 }

151 void I2C::commTaskFunc(void * pvParameters)
152 {
153  I2C * ths = (I2C*) pvParameters;
154 
155  i2c_t * i2c = i2cInit(ths->m_bus, ths->m_SDAGPIO, ths->m_SCLGPIO, I2C_DEFAULT_FREQUENCY);
156  if (!i2c) {
157  ESP_LOGE("unable to init I2C");
158  abort();
159  }
160 
161  i2cFlush(i2c);
162 
163  ths->m_i2c = i2c;
164 
165  // get initial default frequency
166  int freq = i2cGetFrequency(i2c);
167 
168  I2CJobInfo * job = &ths->m_jobInfo;
169 
170  // main send/receive loop
171  while (true) {
172 
173  // unlock waiting task
174  xEventGroupSetBits(ths->m_eventGroup, EVTGROUP_DONE);
175 
176  // wait for another job
177  auto bits = xEventGroupWaitBits(ths->m_eventGroup, EVTGROUP_WRITE | EVTGROUP_READ, true, false, portMAX_DELAY);
178 
179  // setup frequency if necessary
180  if (freq != job->frequency) {
181  freq = job->frequency;
182  i2cSetFrequency(i2c, freq);
183  }
184 
185  if (bits & EVTGROUP_WRITE)
186  job->lastError = i2cWrite(i2c, job->address, job->buffer, job->size, true, job->timeout);
187  else if (bits & EVTGROUP_READ)
188  job->lastError = i2cRead(i2c, job->address, job->buffer, job->size, true, job->timeout, &job->readCount);
189 
190  }
191 
192 }
