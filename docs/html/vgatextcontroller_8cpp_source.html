161 void VGATextController::setResolution(VGATimings const& timings)
162 {
163  // setResolution() already called? If so, stop and free buffers
164  if (m_DMABuffers) {
165  m_GPIOStream.stop();
166  freeBuffers();
167  }
168 
169  m_timings = timings;
170 
171  // inform base class about screen size
172  setScreenSize(m_timings.HVisibleArea, m_timings.VVisibleArea);
173 
174  m_HVSync = packHVSync(false, false);
175 
176  m_DMABuffersCount = 2 * m_timings.VVisibleArea + m_timings.VFrontPorch + m_timings.VSyncPulse + m_timings.VBackPorch;
177 
178  m_DMABuffers = (lldesc_t*) heap_caps_malloc(m_DMABuffersCount * sizeof(lldesc_t), MALLOC_CAP_DMA);
179 
180  m_lines = (uint32_t*) heap_caps_malloc(VGATextController_CHARHEIGHT * VGATextController_WIDTH, MALLOC_CAP_DMA);
181 
182  int rawLineWidth = m_timings.HFrontPorch + m_timings.HSyncPulse + m_timings.HBackPorch + m_timings.HVisibleArea;
183  m_blankLine = (uint8_t*) heap_caps_malloc(rawLineWidth, MALLOC_CAP_DMA);
184  m_syncLine = (uint8_t*) heap_caps_malloc(rawLineWidth, MALLOC_CAP_DMA);
185 
186  // horiz: FRONTPORCH -> SYNC -> BACKPORCH -> VISIBLEAREA
187  //
188  // vert: VISIBLEAREA
189  // FRONTPORCH
190  // SYNC
191  // BACKPORCH
192 
193  for (int i = 0, visLine = 0, invLine = 0; i < m_DMABuffersCount; ++i) {
194 
195  if (i < m_timings.VVisibleArea * 2) {
196 
197  // first part is the same of a blank line
198  m_DMABuffers[i].eof = (visLine == 0 || visLine == VGATextController_CHARHEIGHT / 2 ? 1 : 0);
199  m_DMABuffers[i].sosf = 0;
200  m_DMABuffers[i].offset = 0;
201  m_DMABuffers[i].owner = 1;
202  m_DMABuffers[i].qe.stqe_next = (lldesc_t*) &m_DMABuffers[i + 1];
203  m_DMABuffers[i].length = m_timings.HFrontPorch + m_timings.HSyncPulse + m_timings.HBackPorch;
204  m_DMABuffers[i].size = (m_DMABuffers[i].length + 3) & (~3);
205  m_DMABuffers[i].buf = (uint8_t*) m_blankLine;
206 
207  ++i;
208 
209  // second part is the visible line
210  m_DMABuffers[i].eof = 0;
211  m_DMABuffers[i].sosf = 0;
212  m_DMABuffers[i].offset = 0;
213  m_DMABuffers[i].owner = 1;
214  m_DMABuffers[i].qe.stqe_next = (lldesc_t*) &m_DMABuffers[i + 1];
215  m_DMABuffers[i].length = m_timings.HVisibleArea;
216  m_DMABuffers[i].size = (m_DMABuffers[i].length + 3) & (~3);
217  m_DMABuffers[i].buf = (uint8_t*)(m_lines) + visLine * VGATextController_WIDTH;
218 
219  ++visLine;
220  if (visLine == VGATextController_CHARHEIGHT)
221  visLine = 0;
222 
223  } else {
224 
225  // vertical porchs and sync
226 
227  bool frameResetDesc = (invLine == 0);
228 
229  if (frameResetDesc)
230  s_frameResetDesc = &m_DMABuffers[i];
231 
232  m_DMABuffers[i].eof = (frameResetDesc ? 1 : 0); // prepare for next frame
233  m_DMABuffers[i].sosf = 0;
234  m_DMABuffers[i].offset = 0;
235  m_DMABuffers[i].owner = 1;
236  m_DMABuffers[i].qe.stqe_next = (lldesc_t*) (i == m_DMABuffersCount - 1 ? &m_DMABuffers[0] : &m_DMABuffers[i + 1]);
237  m_DMABuffers[i].length = rawLineWidth;
238  m_DMABuffers[i].size = (m_DMABuffers[i].length + 3) & (~3);
239 
240  if (invLine < m_timings.VFrontPorch || invLine >= m_timings.VFrontPorch + m_timings.VSyncPulse)
241  m_DMABuffers[i].buf = (uint8_t*) m_blankLine;
242  else
243  m_DMABuffers[i].buf = (uint8_t*) m_syncLine;
244 
245  ++invLine;
246 
247  }
248  }
249 
250  fillDMABuffers();
251 
252  s_scanLine = 0;
253 
254  s_blankPatternDWord = m_HVSync | (m_HVSync << 8) | (m_HVSync << 16) | (m_HVSync << 24);
255 
256  if (s_fgbgPattern == nullptr) {
257  s_fgbgPattern = (uint32_t*) heap_caps_malloc(16384, MALLOC_CAP_8BIT | MALLOC_CAP_INTERNAL);
258  for (int i = 0; i < 16; ++i)
259  for (int fg = 0; fg < 16; ++fg)
260  for (int bg = 0; bg < 16; ++bg) {
261  uint8_t fg_pat = preparePixel(RGB222((Color)fg));
262  uint8_t bg_pat = preparePixel(RGB222((Color)bg));
263  s_fgbgPattern[i | (bg << 4) | (fg << 8)] = (i & 0b1000 ? (fg_pat << 16) : (bg_pat << 16)) |
264  (i & 0b0100 ? (fg_pat << 24) : (bg_pat << 24)) |
265  (i & 0b0010 ? (fg_pat << 0) : (bg_pat << 0)) |
266  (i & 0b0001 ? (fg_pat << 8) : (bg_pat << 8));
267  }
268  }
269 
270 
271  // ESP_INTR_FLAG_LEVEL1: should be less than PS2Controller interrupt level, necessary when running on the same core
272  CoreUsage::setBusiestCore(FABGLIB_VIDEO_CPUINTENSIVE_TASKS_CORE);
273  esp_intr_alloc_pinnedToCore(ETS_I2S1_INTR_SOURCE, ESP_INTR_FLAG_LEVEL1 | ESP_INTR_FLAG_IRAM, ISRHandler, this, &m_isr_handle, FABGLIB_VIDEO_CPUINTENSIVE_TASKS_CORE);
274 
275  m_GPIOStream.play(m_timings.frequency, m_DMABuffers);
276 
277  I2S1.int_clr.val = 0xFFFFFFFF;
278  I2S1.int_ena.out_eof = 1;
279 }

292 uint8_t IRAM_ATTR VGATextController::packHVSync(bool HSync, bool VSync)
293 {
294  uint8_t hsync_value = (m_timings.HSyncLogic == '+' ? (HSync ? 1 : 0) : (HSync ? 0 : 1));
295  uint8_t vsync_value = (m_timings.VSyncLogic == '+' ? (VSync ? 1 : 0) : (VSync ? 0 : 1));
296  return (vsync_value << VGA_VSYNC_BIT) | (hsync_value << VGA_HSYNC_BIT);
297 }

306 void VGATextController::fillDMABuffers()
307 {
308  int x = 0;
309  for (; x < m_timings.HFrontPorch; ++x) {
310  VGA_PIXELINROW(m_blankLine, x) = preparePixelWithSync((RGB222){0, 0, 0}, false, false);
311  VGA_PIXELINROW(m_syncLine, x) = preparePixelWithSync((RGB222){0, 0, 0}, false, true);
312  }
313  for (; x < m_timings.HFrontPorch + m_timings.HSyncPulse; ++x) {
314  VGA_PIXELINROW(m_blankLine, x) = preparePixelWithSync((RGB222){0, 0, 0}, true, false);
315  VGA_PIXELINROW(m_syncLine, x) = preparePixelWithSync((RGB222){0, 0, 0}, true, true);
316  }
317  for (; x < m_timings.HFrontPorch + m_timings.HSyncPulse + m_timings.HBackPorch; ++x) {
318  VGA_PIXELINROW(m_blankLine, x) = preparePixelWithSync((RGB222){0, 0, 0}, false, false);
319  VGA_PIXELINROW(m_syncLine, x) = preparePixelWithSync((RGB222){0, 0, 0}, false, true);
320  }
321  int rawLineWidth = m_timings.HFrontPorch + m_timings.HSyncPulse + m_timings.HBackPorch + m_timings.HVisibleArea;
322  for (int rx = 0; x < rawLineWidth; ++x, ++rx) {
323  VGA_PIXELINROW(m_blankLine, x) = preparePixelWithSync((RGB222){0, 0, 0}, false, false);
324  VGA_PIXELINROW(m_syncLine, x) = preparePixelWithSync((RGB222){0, 0, 0}, false, true);
325  for (int i = 0; i < VGATextController_CHARHEIGHT; ++i)
326  VGA_PIXELINROW( ((uint8_t*)(m_lines) + i * VGATextController_WIDTH), rx) = preparePixelWithSync((RGB222){0, 0, 0}, false, false);
327  }
328 }

331 void IRAM_ATTR VGATextController::ISRHandler(void * arg)
332 {
333  #if FABGLIB_VGAXCONTROLLER_PERFORMANCE_CHECK
334  auto s1 = getCycleCount();
335  #endif
336 
337  VGATextController * ctrl = (VGATextController *) arg;
338 
339  if (I2S1.int_st.out_eof && ctrl->m_charData != nullptr) {
340 
341  auto desc = (volatile lldesc_t*) I2S1.out_eof_des_addr;
342 
343  if (desc == s_frameResetDesc) {
344 
345  s_scanLine = 0;
346  s_textRow = 0;
347  s_upperRow = true;
348 
349  if (ctrl->m_cursorEnabled) {
350  ++ctrl->m_cursorCounter;
351  if (ctrl->m_cursorCounter >= ctrl->m_cursorSpeed)
352  ctrl->m_cursorCounter = -ctrl->m_cursorSpeed;
353  }
354 
355  if (ctrl->m_map == nullptr) {
356  I2S1.int_clr.val = I2S1.int_st.val;
357  #if FABGLIB_VGAXCONTROLLER_PERFORMANCE_CHECK
358  s_vgatxtcycles += getCycleCount() - s1;
359  #endif
360  return;
361  }
362 
363  } else if (s_scanLine == 0) {
364  // out of sync, wait for next frame
365  I2S1.int_clr.val = I2S1.int_st.val;
366  #if FABGLIB_VGAXCONTROLLER_PERFORMANCE_CHECK
367  s_vgatxtcycles += getCycleCount() - s1;
368  #endif
369  return;
370  }
371 
372  int scanLine = s_scanLine;
373 
374  const int lineIndex = scanLine % VGATextController_CHARHEIGHT;
375 
376  auto lines = ctrl->m_lines;
377 
378  if (s_textRow < ctrl->m_rows) {
379 
380  int cursorCol = 0;
381  int cursorFGBG = 0;
382  const auto cursorVisible = (ctrl->m_cursorEnabled && ctrl->m_cursorCounter >= 0 && s_textRow == ctrl->m_cursorRow);
383  if (cursorVisible) {
384  cursorCol = ctrl->m_cursorCol;
385  cursorFGBG = (ctrl->m_cursorForeground << 4) | (ctrl->m_cursorBackground << 8);
386  }
387 
388  const auto charData = ctrl->m_charData + (s_upperRow ? 0 : VGATextController_CHARHEIGHT / 2);
389  auto mapItemPtr = ctrl->m_map + s_textRow * VGATextController_COLUMNS;
390 
391  for (int col = 0; col < VGATextController_COLUMNS; ++col, ++mapItemPtr) {
392 
393  const auto mapItem = *mapItemPtr;
394 
395  int fgbg = (mapItem >> 4) & 0b111111110000;
396 
397  const auto options = glyphMapItem_getOptions(mapItem);
398 
399  // invert?
400  if (options.invert)
401  fgbg = ((fgbg >> 4) & 0b11110000) | ((fgbg << 4) & 0b111100000000);
402 
403  // cursor?
404  if (cursorVisible && col == cursorCol)
405  fgbg = cursorFGBG;
406 
407  uint32_t * dest = lines + lineIndex * VGATextController_WIDTH / sizeof(uint32_t) + col * VGATextController_CHARWIDTHBYTES * 2;
408 
409  // blank?
410  if (options.blank) {
411 
412  for (int rowInChar = 0; rowInChar < VGATextController_CHARHEIGHT / 2; ++rowInChar) {
413  int v = s_fgbgPattern[fgbg];
414  *dest = v;
415  *(dest + 1) = v;
416  dest += VGATextController_WIDTH / sizeof(uint32_t);
417  }
418 
419  } else {
420 
421  const bool underline = (s_upperRow == false && options.underline);
422  const bool bold = options.bold;
423 
424  auto charRowPtr = charData + glyphMapItem_getIndex(mapItem) * VGATextController_CHARHEIGHT * VGATextController_CHARWIDTHBYTES;
425 
426  for (int rowInChar = 0; rowInChar < VGATextController_CHARHEIGHT / 2; ++rowInChar) {
427  auto charRowData = *charRowPtr;
428 
429  // bold?
430  if (bold)
431  charRowData |= charRowData >> 1;
432 
433  *dest = s_fgbgPattern[(charRowData >> 4) | fgbg];
434  *(dest + 1) = s_fgbgPattern[(charRowData & 0xF) | fgbg];
435 
436  dest += VGATextController_WIDTH / sizeof(uint32_t);
437  charRowPtr += VGATextController_CHARWIDTHBYTES;
438  }
439 
440  // underline?
441  if (underline) {
442  dest -= VGATextController_WIDTH / sizeof(uint32_t);
443  uint32_t v = s_fgbgPattern[0xF | fgbg];
444  *dest = v;
445  *(dest + 1) = v;
446  }
447 
448  }
449 
450  }
451 
452  if (s_upperRow) {
453  s_upperRow = false;
454  } else {
455  s_upperRow = true;
456  ++s_textRow;
457  }
458 
459  } else {
460  for (int i = 0; i < VGATextController_CHARHEIGHT / 2; ++i) {
461  auto dest = lines + ((scanLine + i) % VGATextController_CHARHEIGHT) * VGATextController_WIDTH / sizeof(uint32_t);
462  for (int i = 0; i < VGATextController_COLUMNS; ++i) {
463  *dest++ = s_blankPatternDWord;
464  *dest++ = s_blankPatternDWord;
465  }
466  }
467  }
468 
469  scanLine += VGATextController_CHARHEIGHT / 2;
470 
471  s_scanLine = scanLine;
472 
473  }
474 
475  #if FABGLIB_VGAXCONTROLLER_PERFORMANCE_CHECK
476  s_vgatxtcycles += getCycleCount() - s1;
477  #endif
478 
479  I2S1.int_clr.val = I2S1.int_st.val;
480 }
