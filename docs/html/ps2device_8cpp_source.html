141 int PS2Device::getData(int timeOutMS)
142 {
143  constexpr int INTER_GETDATA_TIMEOUT_MS = 100;
144  constexpr int INTER_GETDATA_PAUSE_MS = 10;
145 
146  int interTimeOut = timeOutMS > -1 ? imin(timeOutMS, INTER_GETDATA_TIMEOUT_MS) : INTER_GETDATA_TIMEOUT_MS;
147 
148  int ret = -1;
149  TimeOut timeout;
150  while (true) {
151  lock(-1);
152  ret = PS2Controller::getData(m_PS2Port, interTimeOut);
153  unlock();
154  if (ret > -1 || parityError() || syncError() || CLKTimeOutError() || timeout.expired(timeOutMS))
155  break;
156  // give the opportunity for other sends
157  vTaskDelay(INTER_GETDATA_PAUSE_MS / portTICK_PERIOD_MS);
158  }
159  return ret;
160 }

163 bool PS2Device::sendCommand(uint8_t cmd, uint8_t expectedReply)
164 {
165  constexpr int INTER_WAITREPLY_TIMEOUT_MS = 10;
166 
167  PS2DeviceLock deviceLock(this);
168 
169  // temporary disable RX for the other port
170  PS2PortAutoDisableRX autoDisableRX(!m_PS2Port);
171 
172  PS2Controller::sendData(cmd, m_PS2Port);
173  TimeOut timeout;
174  do {
175  if (PS2Controller::getData(m_PS2Port, INTER_WAITREPLY_TIMEOUT_MS) == expectedReply)
176  return true;
177  } while (!timeout.expired(m_cmdTimeOut));
178  return false;
179 }

228 // return value is always valid
229 bool PS2Device::send_cmdIdentify(PS2DeviceType * result)
230 {
231  PS2DeviceLock deviceLock(this);
232  *result = PS2DeviceType::UnknownPS2Device;
233  if (!send_cmdDisableScanning())
234  return false;
235  if (!sendCommand(PS2_CMD_IDENTIFY, PS2_REPLY_ACK))
236  return false;
237  int b1 = getData(m_cmdTimeOut);
238  int b2 = getData(m_cmdTimeOut);
239  m_deviceID = (uint8_t)b1 | ((uint8_t)b2 << 8);
240  if (b1 == -1 && b2 == -1)
241  *result = PS2DeviceType::OldATKeyboard;
242  else if (b1 == 0x00 && b2 == -1)
243  *result = PS2DeviceType::MouseStandard;
244  else if (b1 == 0x03 && b2 == -1)
245  *result = PS2DeviceType::MouseWithScrollWheel;
246  else if (b1 == 0x04 && b2 == -1)
247  *result = PS2DeviceType::Mouse5Buttons;
248  else if ((b1 == 0xAB && b2 == 0x41) || (b1 == 0xAB && b2 == 0xC1))
249  *result = PS2DeviceType::MF2KeyboardWithTranslation;
250  else if (b1 == 0xAB && b2 == 0x83)
251  *result = PS2DeviceType::M2Keyboard;
252  return send_cmdEnableScanning();
253 }

273 // repeatRateMS : 33 ms ... 500 ms (in steps as above REPEATRATES table)
274 // repeatDelayMS : 250 ms ... 1000 ms (in steps of 250 ms)
275 bool PS2Device::send_cmdTypematicRateAndDelay(int repeatRateMS, int repeatDelayMS)
276 {
277  PS2DeviceLock deviceLock(this);
278  if (!sendCommand(PS2_CMD_SET_TYPEMATIC_RATE_AND_DELAY, PS2_REPLY_ACK))
279  return false;
280  uint8_t byteToSend = 0b01011; // default repeat rate 10.9 characters per seconds (91ms)
281  for (int i = 0; i < 32; ++i)
282  if (REPEATRATES[i] >= repeatRateMS) {
283  byteToSend = i;
284  break;
285  }
286  byteToSend |= (repeatDelayMS / 250 - 1) << 5;
287  return sendCommand(byteToSend, PS2_REPLY_ACK);
288 }
