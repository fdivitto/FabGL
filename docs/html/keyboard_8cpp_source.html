104 // reset keyboard, set scancode 2 and US layout
105 bool Keyboard::reset()
106 {
107  memset(m_VKMap, 0, sizeof(m_VKMap));
108 
109  // sets default layout
110  setLayout(&USLayout);
111 
112  // 350ms keyboard poweron delay (look at NXP M68HC08 designer reference manual)
113  vTaskDelay(350 / portTICK_PERIOD_MS);
114 
115  // tries up to three times to reset keyboard
116  for (int i = 0; i < 3; ++i) {
117  m_keyboardAvailable = send_cmdReset();
118  if (m_keyboardAvailable)
119  break;
120  vTaskDelay(350 / portTICK_PERIOD_MS);
121  }
122  // give the time to the device to be fully initialized
123  vTaskDelay(200 / portTICK_PERIOD_MS);
124 
125  send_cmdSetScancodeSet(2);
126 
127  return m_keyboardAvailable;
128 }

131 bool Keyboard::setScancodeSet(int value)
132 {
133  if (m_SCodeToVKConverterTask) {
134  // virtual keys enabled, just 1 and 2 are allowed
135  if (value == 1 || value == 2) {
136  m_scancodeSet = value;
137  return true;
138  }
139  } else {
140  // no virtual keys enabled, just try to tell keyboard which set we need
141  if (send_cmdSetScancodeSet(value)) {
142  m_scancodeSet = value;
143  return true;
144  }
145  }
146  return false;
147 }

182 int Keyboard::getNextScancode(int timeOutMS, bool requestResendOnTimeOut)
183 {
184  while (true) {
185  int r = getData(timeOutMS);
186  if (r == -1 && CLKTimeOutError()) {
187  // try to recover a stall sending a re-enable scanning command
188  send_cmdEnableScanning();
189  }
190  if (r == -1 && requestResendOnTimeOut) {
191  requestToResendLastByte();
192  continue;
193  }
194  return r;
195  }

233 {
234  switch (virtualKey) {
235  case VK_SPACE:
236  return m_CTRL ? ASCII_NUL : ASCII_SPC; // CTRL SPACE = NUL, otherwise 0x20
237 
238  case VK_0 ... VK_9:
239  if (m_CTRL) {
240  switch (virtualKey) {
241  case VK_2:
242  return ASCII_NUL; // CTRL + 2 = NUL
243  case VK_6:
244  return ASCII_RS; // CTRL + 6 = RS
245  default:
246  return -1;
247  }
248  }
249  // otherwise digits
250  return virtualKey - VK_0 + '0';
251 
252  case VK_KP_0 ... VK_KP_9:

557 VirtualKey Keyboard::scancodeToVK(uint8_t scancode, bool isExtended, KeyboardLayout const * layout)
558 {
559  VirtualKey vk = VK_NONE;
560 
561  if (layout == nullptr)
562  layout = m_layout;
563 
564  VirtualKeyDef const * def = isExtended ? layout->exScancodeToVK : layout->scancodeToVK;
565  for (; def->scancode; ++def)
566  if (def->scancode == scancode) {
567  vk = def->virtualKey;
568  break;
569  }
570 
571  if (vk == VK_NONE && layout->inherited)
572  vk = scancodeToVK(scancode, isExtended, layout->inherited);
573 
574  // manage keypad
575  // NUMLOCK ON, SHIFT OFF => generate VK_KP_number
576  // NUMLOCK ON, SHIFT ON => generate VK_KP_cursor_control (as when NUMLOCK is OFF)
577  // NUMLOCK OFF => generate VK_KP_cursor_control
578  if (m_NUMLOCK & !m_SHIFT) {
579  switch (vk) {
580  case VK_KP_DELETE:
581  vk = VK_KP_PERIOD;
582  break;
583  case VK_KP_INSERT:
584  vk = VK_KP_0;
585  break;
586  case VK_KP_END:
587  vk = VK_KP_1;
588  break;
589  case VK_KP_DOWN:
590  vk = VK_KP_2;
591  break;
592  case VK_KP_PAGEDOWN:
593  vk = VK_KP_3;
594  break;
595  case VK_KP_LEFT:
596  vk = VK_KP_4;
597  break;
598  case VK_KP_CENTER:
599  vk = VK_KP_5;
600  break;
601  case VK_KP_RIGHT:
602  vk = VK_KP_6;
603  break;
604  case VK_KP_HOME:
605  vk = VK_KP_7;
606  break;
607  case VK_KP_UP:
608  vk = VK_KP_8;
609  break;
610  case VK_KP_PAGEUP:
611  vk = VK_KP_9;
612  break;
613  default:
614  break;
615  }
616  }
617 
618  return vk;
619 }

622 VirtualKey Keyboard::manageCAPSLOCK(VirtualKey vk)
623 {
624  if (m_CAPSLOCK) {
625  // inverts letters case
626  if (vk >= VK_a && vk <= VK_z)
627  vk = (VirtualKey)(vk - VK_a + VK_A);
628  else if (vk >= VK_A && vk <= VK_Z)
629  vk = (VirtualKey)(vk - VK_A + VK_a);
630  }
631  return vk;
632 }

635 VirtualKey Keyboard::VKtoAlternateVK(VirtualKey in_vk, bool down, KeyboardLayout const * layout)
636 {
637  VirtualKey vk = VK_NONE;
638 
639  if (layout == nullptr)
640  layout = m_layout;
641 
642  // this avoids releasing a required key when SHIFT has been pressed after the key but before releasing
643  if (!down && isVKDown(in_vk))
644  vk = in_vk;
645 
646  if (vk == VK_NONE) {
647  // handle this case:
648  // - derived KEY up without any SHIFT (because released before the KEY, ie SHIFT+"1" => "!", but you release the SHIFT before "1")

651  if (def->reqVirtualKey == in_vk && isVKDown(def->virtualKey)) {
652  vk = def->virtualKey;
653  break;
654  }
655  }
656  }
657

660  // - KEY down with SHIFTs already down
661  // - KEY up with SHIFTs still down
662  for (AltVirtualKeyDef const * def = layout->alternateVK; def->reqVirtualKey != VK_NONE; ++def) {
663  if (def->reqVirtualKey == in_vk && def->ctrl == m_CTRL &&
664  def->lalt == m_LALT &&
665  def->ralt == m_RALT &&
666  def->shift == m_SHIFT) {
667  vk = def->virtualKey;
668  break;
669  }
670  }
671  }
672

674  vk = VKtoAlternateVK(in_vk, down, layout->inherited);

680 bool Keyboard::blockingGetVirtualKey(VirtualKeyItem * item)
681 {
682  item->vk = VK_NONE;
683  item->down = true;
684  item->CTRL = m_CTRL;
685  item->LALT = m_LALT;
686  item->RALT = m_RALT;
687  item->SHIFT = m_SHIFT;
688  item->GUI = m_GUI;
689  item->CAPSLOCK = m_CAPSLOCK;
690  item->NUMLOCK = m_NUMLOCK;
691 
692  uint8_t * scode = item->scancode;
693 
694  *scode = getNextScancode();
695  if (*scode == 0xE0) {
696  // two bytes scancode
697  *(++scode) = getNextScancode(100, true);
698  if (*scode == 0xF0) {
699  // two bytes scancode key up
700  *(++scode) = getNextScancode(100, true);
701  item->vk = scancodeToVK(*scode, true);
702  item->down = false;
703  } else {
704  // two bytes scancode key down
705  item->vk = scancodeToVK(*scode, true);
706  }
707  } else if (*scode == 0xE1) {
708  // special case: "PAUSE" : 0xE1, 0x14, 0x77, 0xE1, 0xF0, 0x14, 0xF0, 0x77
709  static const uint8_t PAUSECODES[] = {0x14, 0x77, 0xE1, 0xF0, 0x14, 0xF0, 0x77};
710  for (int i = 0; i < sizeof(PAUSECODES); ++i) {
711  *(++scode) = getNextScancode(100, true);
712  if (*scode != PAUSECODES[i])
713  break;
714  else if (i == sizeof(PAUSECODES) - 1)
715  item->vk = VK_PAUSE;
716  }
717  } else if (*scode == 0xF0) {
718  // one byte scancode, key up
719  *(++scode) = getNextScancode(100, true);
720  item->vk = scancodeToVK(*scode, false);
721  item->down = false;
722  } else {
723  // one byte scancode, key down
724  item->vk = scancodeToVK(*scode, false);
725  }
726 
727  if (item->vk != VK_NONE) {
728 
729  // manage CAPSLOCK
730  item->vk = manageCAPSLOCK(item->vk);
731 
732  // alternate VK (virtualkeys modified by shift, alt, ...)
733  item->vk = VKtoAlternateVK(item->vk, item->down);
734 
735  // update shift, alt, ctrl, capslock, numlock and scrollock states and LEDs
736  switch (item->vk) {
737  case VK_LCTRL:
738  case VK_RCTRL:
739  m_CTRL = item->down;
740  break;
741  case VK_LALT:
742  m_LALT = item->down;
743  break;
744  case VK_RALT:
745  m_RALT = item->down;
746  break;
747  case VK_LSHIFT:
748  case VK_RSHIFT:
749  m_SHIFT = item->down;
750  break;
751  case VK_LGUI:
752  case VK_RGUI:
753  m_GUI = item->down;
754  break;
755  case VK_CAPSLOCK:
756  if (!item->down) {
757  m_CAPSLOCK = !m_CAPSLOCK;
758  updateLEDs();
759  }
760  break;
761  case VK_NUMLOCK:
762  if (!item->down) {
763  m_NUMLOCK = !m_NUMLOCK;
764  updateLEDs();
765  }
766  break;
767  case VK_SCROLLLOCK:
768  if (!item->down) {
769  m_SCROLLLOCK = !m_SCROLLLOCK;
770  updateLEDs();
771  }
772  break;
773  default:
774  break;
775  }
776 
777  }
778 
779  // manage dead keys - Implemented by Carles Oriol (https://github.com/carlesoriol)
780  for (VirtualKey const * dk = m_layout->deadKeysVK; *dk != VK_NONE; ++dk) {
781  if (item->vk == *dk) {
782  m_lastDeadKey = item->vk;
783  item->vk = VK_NONE;
784  }
785  }
786  if (item->vk != m_lastDeadKey && item->vk != VK_NONE) {
787  for (DeadKeyVirtualKeyDef const * dk = m_layout->deadkeysToVK; dk->deadKey != VK_NONE; ++dk) {
788  if (item->vk == dk->reqVirtualKey && m_lastDeadKey == dk->deadKey) {
789  item->vk = dk->virtualKey;
790  break;
791  }
792  }
793  if (!item->down && (item->vk != m_lastDeadKey) && (item->vk != VK_RSHIFT) && (item->vk != VK_LSHIFT))
794  m_lastDeadKey = VK_NONE;
795  }
796 
797  // ending zero to item->scancode
798  if (scode < item->scancode + sizeof(VirtualKeyItem::scancode) - 1)
799  *(++scode) = 0;
800 
801  // fill ASCII field
802  int ascii = virtualKeyToASCII(item->vk);
803  item->ASCII = ascii > -1 ? ascii : 0;
804 
805  return item->vk != VK_NONE;
806 }

908 void Keyboard::SCodeToVKConverterTask(void * pvParameters)
909 {
910  Keyboard * keyboard = (Keyboard*) pvParameters;
911 
912  // manage ALT + Keypad num
913  uint8_t ALTNUMValue = 0; // current value (0 = no value, 0 is not allowed)
914 
915  while (true) {
916 
917  VirtualKeyItem item;
918 
919  if (keyboard->blockingGetVirtualKey(&item)) {
920 
921  // onVirtualKey may set item.vk = VK_NONE!
922  keyboard->onVirtualKey(&item.vk, item.down);
923 
924  if (item.vk != VK_NONE) {
925 
926  // manage left-ALT + NUM
927  if (!isALT(item.vk) && keyboard->m_LALT) {
928  // ALT was down, is this a keypad number?
929  int num = convKeypadVKToNum(item.vk);
930  if (num >= 0) {
931  // yes this is a keypad num, if down update ALTNUMValue
932  if (item.down)
933  ALTNUMValue = (ALTNUMValue * 10 + num) & 0xff;
934  } else {
935  // no, back to normal case
936  ALTNUMValue = 0;
937  keyboard->postVirtualKeyItem(item);
938  }
939  } else if (ALTNUMValue > 0 && isALT(item.vk) && !item.down) {
940  // ALT is up and ALTNUMValue contains a valid value, add it
941  keyboard->postVirtualKeyItem(item); // post ALT up
942  item.vk = VK_ASCII;
943  item.down = true;
944  item.scancode[0] = 0;
945  item.ASCII = ALTNUMValue;
946  keyboard->postVirtualKeyItem(item); // ascii key down
947  item.down = false;
948  keyboard->postVirtualKeyItem(item); // ascii key up
949  ALTNUMValue = 0;
950  } else {
951  // normal case
952  keyboard->postVirtualKeyItem(item);
953  }
954 
955  }
956 
957  }
958 
959  }
