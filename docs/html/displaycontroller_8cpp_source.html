73 // 0 .. 63 => 0
74 // 64 .. 127 => 1
75 // 128 .. 191 => 2
76 // 192 .. 255 => 3
77 RGB222::RGB222(RGB888 const & value)
78 {
79  if (lowBitOnly) {
80  R = value.R ? 3 : 0;
81  G = value.G ? 3 : 0;
82  B = value.B ? 3 : 0;
83  } else {
84  R = value.R >> 6;
85  G = value.G >> 6;
86  B = value.B >> 6;
87  }
88 }

108 uint8_t RGB888toPackedRGB222(RGB888 const & rgb)
109 {
110  // 64 colors
111  static const int CONVR64[4] = { 0 << 0, // 00XXXXXX (0..63)
112  1 << 0, // 01XXXXXX (64..127)
113  2 << 0, // 10XXXXXX (128..191)
114  3 << 0, }; // 11XXXXXX (192..255)
115  static const int CONVG64[4] = { 0 << 2, // 00XXXXXX (0..63)
116  1 << 2, // 01XXXXXX (64..127)
117  2 << 2, // 10XXXXXX (128..191)
118  3 << 2, }; // 11XXXXXX (192..255)
119  static const int CONVB64[4] = { 0 << 4, // 00XXXXXX (0..63)
120  1 << 4, // 01XXXXXX (64..127)
121  2 << 4, // 10XXXXXX (128..191)
122  3 << 4, }; // 11XXXXXX (192..255)
123  // 8 colors
124  static const int CONVR8[4] = { 0 << 0, // 00XXXXXX (0..63)
125  3 << 0, // 01XXXXXX (64..127)
126  3 << 0, // 10XXXXXX (128..191)
127  3 << 0, }; // 11XXXXXX (192..255)
128  static const int CONVG8[4] = { 0 << 2, // 00XXXXXX (0..63)
129  3 << 2, // 01XXXXXX (64..127)
130  3 << 2, // 10XXXXXX (128..191)
131  3 << 2, }; // 11XXXXXX (192..255)
132  static const int CONVB8[4] = { 0 << 4, // 00XXXXXX (0..63)
133  3 << 4, // 01XXXXXX (64..127)
134  3 << 4, // 10XXXXXX (128..191)
135  3 << 4, }; // 11XXXXXX (192..255)
136 
137  if (RGB222::lowBitOnly)
138  return (CONVR8[rgb.R >> 6]) | (CONVG8[rgb.G >> 6]) | (CONVB8[rgb.B >> 6]);
139  else
140  return (CONVR64[rgb.R >> 6]) | (CONVG64[rgb.G >> 6]) | (CONVB64[rgb.B >> 6]);
141 }

263 void Bitmap::allocate()
264 {
265  if (dataAllocated) {
266  free((void*)data);
267  data = nullptr;
268  }
269  dataAllocated = true;
270  switch (format) {
271  case PixelFormat::Undefined:
272  case PixelFormat::Native:
273  break;
274  case PixelFormat::Mask:
275  data = (uint8_t*) heap_caps_malloc((width + 7) / 8 * height, MALLOC_CAP_8BIT | MALLOC_CAP_INTERNAL);
276  break;
277  case PixelFormat::RGBA2222:
278  data = (uint8_t*) heap_caps_malloc(width * height, MALLOC_CAP_8BIT | MALLOC_CAP_INTERNAL);
279  break;
280  case PixelFormat::RGBA8888:
281  data = (uint8_t*) heap_caps_malloc(width * height * 4, MALLOC_CAP_8BIT | MALLOC_CAP_INTERNAL);
282  break;
283  }
284 }

308 {
309  int rowlen = (width + 7) / 8;
310  uint8_t * rowptr = data + y * rowlen;
311  if (value)
312  rowptr[x >> 3] |= 0x80 >> (x & 7);
313  else
314  rowptr[x >> 3] &= ~(0x80 >> (x & 7));

330 int Bitmap::getAlpha(int x, int y)
331 {
332  int r = 0;
333  switch (format) {
334  case PixelFormat::Undefined:
335  break;
336  case PixelFormat::Native:
337  r = 0xff;
338  break;
339  case PixelFormat::Mask:
340  {
341  int rowlen = (width + 7) / 8;
342  uint8_t * rowptr = data + y * rowlen;
343  r = (rowptr[x >> 3] >> (7 - (x & 7))) & 1;
344  break;
345  }
346  case PixelFormat::RGBA2222:
347  r = ((RGBA2222*)data)[y * height + x].A;
348  break;
349  case PixelFormat::RGBA8888:
350  r = ((RGBA8888*)data)[y * height + x].A;
351  break;
352  }
353  return r;
354 }

419 void BitmappedDisplayController::addPrimitive(Primitive & primitive)
420 {
421  if ((m_backgroundPrimitiveExecutionEnabled && m_doubleBuffered == false) || primitive.cmd == PrimitiveCmd::SwapBuffers) {
422  primitiveReplaceDynamicBuffers(primitive);
423  xQueueSendToBack(m_execQueue, &primitive, portMAX_DELAY);
424 
425  if (m_doubleBuffered) {
426  // wait notufy from PrimitiveCmd::SwapBuffers executor
427  ulTaskNotifyTake(true, portMAX_DELAY);
428  }
429 
430  } else {
431  Rect updateRect = Rect(SHRT_MAX, SHRT_MAX, SHRT_MIN, SHRT_MIN);
432  execPrimitive(primitive, updateRect, false);
433  showSprites(updateRect);
434  }
435 }

438 // some primitives require additional buffers (like drawPath and fillPath).
439 // this function copies primitive data into an allocated buffer (using LightMemoryPool allocator) that
440 // will be released inside primitive drawing code.
441 void BitmappedDisplayController::primitiveReplaceDynamicBuffers(Primitive & primitive)
442 {
443  switch (primitive.cmd) {
444  case PrimitiveCmd::DrawPath:
445  case PrimitiveCmd::FillPath:
446  {
447  int sz = primitive.path.pointsCount * sizeof(Point);
448  if (sz < FABGLIB_PRIMITIVES_DYNBUFFERS_SIZE) {
449  void * newbuf = nullptr;
450  // wait until we have enough free space
451  while ((newbuf = m_primDynMemPool.alloc(sz)) == nullptr)
452  taskYIELD();
453  memcpy(newbuf, primitive.path.points, sz);
454  primitive.path.points = (Point*)newbuf;
455  primitive.path.freePoints = true;
456  }
457  break;
458  }
459 
460  default:
461  break;
462  }
463 }

500 {

513 // Use for fast queue processing. Warning, may generate flickering because don't care of vertical sync
514 // Do not call inside ISR
515 void IRAM_ATTR BitmappedDisplayController::processPrimitives()
516 {
517  suspendBackgroundPrimitiveExecution();
518  Rect updateRect = Rect(SHRT_MAX, SHRT_MAX, SHRT_MIN, SHRT_MIN);
519  Primitive prim;
520  while (xQueueReceive(m_execQueue, &prim, 0) == pdTRUE)
521  execPrimitive(prim, updateRect, false);
522  showSprites(updateRect);
523  resumeBackgroundPrimitiveExecution();
524  Primitive p(PrimitiveCmd::Refresh, updateRect);
525  addPrimitive(p);
526 }

529 void BitmappedDisplayController::setSprites(Sprite * sprites, int count, int spriteSize)
530 {
531  processPrimitives();
532  primitivesExecutionWait();
533  m_sprites = sprites;
534  m_spriteSize = spriteSize;
535  m_spritesCount = count;
536 
537  // allocates background buffer
538  if (!isDoubleBuffered()) {
539  uint8_t * spritePtr = (uint8_t*)m_sprites;
540  for (int i = 0; i < m_spritesCount; ++i, spritePtr += m_spriteSize) {
541  Sprite * sprite = (Sprite*) spritePtr;
542  int reqBackBufferSize = 0;
543  for (int i = 0; i < sprite->framesCount; ++i)
544  reqBackBufferSize = tmax(reqBackBufferSize, sprite->frames[i]->width * getBitmapSavePixelSize() * sprite->frames[i]->height);
545  if (reqBackBufferSize > 0)
546  sprite->savedBackground = (uint8_t*) realloc(sprite->savedBackground, reqBackBufferSize);
547  }
548  }
549 }

565 void IRAM_ATTR BitmappedDisplayController::hideSprites(Rect & updateRect)
566 {
567  if (!m_spritesHidden) {
568  m_spritesHidden = true;
569 
570  // normal sprites
571  if (spritesCount() > 0 && !isDoubleBuffered()) {
572  // restore saved backgrounds
573  for (int i = spritesCount() - 1; i >= 0; --i) {
574  Sprite * sprite = getSprite(i);
575  if (sprite->allowDraw && sprite->savedBackgroundWidth > 0) {
576  int savedX = sprite->savedX;
577  int savedY = sprite->savedY;
578  int savedWidth = sprite->savedBackgroundWidth;
579  int savedHeight = sprite->savedBackgroundHeight;
580  Bitmap bitmap(savedWidth, savedHeight, sprite->savedBackground, PixelFormat::Native);
581  absDrawBitmap(savedX, savedY, &bitmap, nullptr, true);
582  updateRect = updateRect.merge(Rect(savedX, savedY, savedX + savedWidth - 1, savedY + savedHeight - 1));
583  sprite->savedBackgroundWidth = sprite->savedBackgroundHeight = 0;
584  }
585  }
586  }
587 
588  // mouse cursor sprite
589  Sprite * mouseSprite = mouseCursor();
590  if (mouseSprite->savedBackgroundWidth > 0) {
591  int savedX = mouseSprite->savedX;
592  int savedY = mouseSprite->savedY;
593  int savedWidth = mouseSprite->savedBackgroundWidth;
594  int savedHeight = mouseSprite->savedBackgroundHeight;
595  Bitmap bitmap(savedWidth, savedHeight, mouseSprite->savedBackground, PixelFormat::Native);
596  absDrawBitmap(savedX, savedY, &bitmap, nullptr, true);
597  updateRect = updateRect.merge(Rect(savedX, savedY, savedX + savedWidth - 1, savedY + savedHeight - 1));
598  mouseSprite->savedBackgroundWidth = mouseSprite->savedBackgroundHeight = 0;
599  }
600 
601  }
602 }

605 void IRAM_ATTR BitmappedDisplayController::showSprites(Rect & updateRect)
606 {
607  if (m_spritesHidden) {
608  m_spritesHidden = false;
609 
610  // normal sprites
611  // save backgrounds and draw sprites
612  for (int i = 0; i < spritesCount(); ++i) {
613  Sprite * sprite = getSprite(i);
614  if (sprite->visible && sprite->allowDraw && sprite->getFrame()) {
615  // save sprite X and Y so other threads can change them without interferring
616  int spriteX = sprite->x;
617  int spriteY = sprite->y;
618  Bitmap const * bitmap = sprite->getFrame();
619  int bitmapWidth = bitmap->width;
620  int bitmapHeight = bitmap->height;
621  absDrawBitmap(spriteX, spriteY, bitmap, sprite->savedBackground, true);
622  sprite->savedX = spriteX;
623  sprite->savedY = spriteY;
624  sprite->savedBackgroundWidth = bitmapWidth;
625  sprite->savedBackgroundHeight = bitmapHeight;
626  if (sprite->isStatic)
627  sprite->allowDraw = false;
628  updateRect = updateRect.merge(Rect(spriteX, spriteY, spriteX + bitmapWidth - 1, spriteY + bitmapHeight - 1));
629  }
630  }
631 
632  // mouse cursor sprite
633  // save backgrounds and draw mouse cursor
634  Sprite * mouseSprite = mouseCursor();
635  if (mouseSprite->visible && mouseSprite->getFrame()) {
636  // save sprite X and Y so other threads can change them without interferring
637  int spriteX = mouseSprite->x;
638  int spriteY = mouseSprite->y;
639  Bitmap const * bitmap = mouseSprite->getFrame();
640  int bitmapWidth = bitmap->width;
641  int bitmapHeight = bitmap->height;
642  absDrawBitmap(spriteX, spriteY, bitmap, mouseSprite->savedBackground, true);
643  mouseSprite->savedX = spriteX;
644  mouseSprite->savedY = spriteY;
645  mouseSprite->savedBackgroundWidth = bitmapWidth;
646  mouseSprite->savedBackgroundHeight = bitmapHeight;
647  updateRect = updateRect.merge(Rect(spriteX, spriteY, spriteX + bitmapWidth - 1, spriteY + bitmapHeight - 1));
648  }
649 
650  }
651 }

693 void IRAM_ATTR BitmappedDisplayController::execPrimitive(Primitive const & prim, Rect & updateRect, bool insideISR)
694 {
695  switch (prim.cmd) {
696  case PrimitiveCmd::Flush:
697  break;
698  case PrimitiveCmd::Refresh:
699  updateRect = updateRect.merge(prim.rect);
700  break;
701  case PrimitiveCmd::Reset:
702  resetPaintState();
703  break;
704  case PrimitiveCmd::SetPenColor:
705  paintState().penColor = prim.color;
706  break;
707  case PrimitiveCmd::SetBrushColor:
708  paintState().brushColor = prim.color;
709  break;
710  case PrimitiveCmd::SetPixel:
711  setPixelAt( (PixelDesc) { prim.position, getActualPenColor() }, updateRect );
712  break;
713  case PrimitiveCmd::SetPixelAt:
714  setPixelAt(prim.pixelDesc, updateRect);
715  break;
716  case PrimitiveCmd::MoveTo:
717  paintState().position = Point(prim.position.X + paintState().origin.X, prim.position.Y + paintState().origin.Y);
718  break;
719  case PrimitiveCmd::LineTo:
720  lineTo(prim.position, updateRect);
721  break;
722  case PrimitiveCmd::FillRect:
723  fillRect(prim.rect, getActualBrushColor(), updateRect);
724  break;
725  case PrimitiveCmd::DrawRect:
726  drawRect(prim.rect, updateRect);
727  break;
728  case PrimitiveCmd::FillEllipse:

898 // McIlroy's algorithm
899 void IRAM_ATTR BitmappedDisplayController::fillEllipse(int centerX, int centerY, Size const & size, RGB888 const & color, Rect & updateRect)
900 {
901  const int clipX1 = paintState().absClippingRect.X1;
902  const int clipY1 = paintState().absClippingRect.Y1;
903  const int clipX2 = paintState().absClippingRect.X2;
904  const int clipY2 = paintState().absClippingRect.Y2;
905 
906  const int halfWidth = size.width / 2;
907  const int halfHeight = size.height / 2;
908 
909  updateRect = updateRect.merge(Rect(centerX - halfWidth, centerY - halfHeight, centerX + halfWidth, centerY + halfHeight));
910  hideSprites(updateRect);
911 
912  const int a2 = halfWidth * halfWidth;
913  const int b2 = halfHeight * halfHeight;
914  const int crit1 = -(a2 / 4 + halfWidth % 2 + b2);
915  const int crit2 = -(b2 / 4 + halfHeight % 2 + a2);
916  const int crit3 = -(b2 / 4 + halfHeight % 2);
917  const int d2xt = 2 * b2;
918  const int d2yt = 2 * a2;
919  int x = 0; // travels from 0 up to halfWidth
920  int y = halfHeight; // travels from halfHeight down to 0
921  int width = 1;
922  int t = -a2 * y;
923  int dxt = 2 * b2 * x;
924  int dyt = -2 * a2 * y;
925 
926  while (y >= 0 && x <= halfWidth) {
927  if (t + b2 * x <= crit1 || t + a2 * y <= crit3) {
928  x++;
929  dxt += d2xt;
930  t += dxt;
931  width += 2;
932  } else {
933  int col1 = centerX - x;
934  int col2 = centerX - x + width - 1;
935  if (col1 <= clipX2 && col2 >= clipX1) {
936  col1 = iclamp(col1, clipX1, clipX2);
937  col2 = iclamp(col2, clipX1, clipX2);
938  int row1 = centerY - y;
939  int row2 = centerY + y;
940  if (row1 >= clipY1 && row1 <= clipY2)
941  rawFillRow(row1, col1, col2, color);
942  if (y != 0 && row2 >= clipY1 && row2 <= clipY2)
943  rawFillRow(row2, col1, col2, color);
944  }
945  if (t - a2 * y <= crit2) {
946  x++;
947  dxt += d2xt;
948  t += dxt;
949  width += 2;
950  }
951  y--;
952  dyt += d2yt;
953  t += dyt;
954  }
955  }
956  // one line horizontal ellipse case
957  if (halfHeight == 0 && centerY >= clipY1 && centerY <= clipY2)
958  rawFillRow(centerY, iclamp(centerX - halfWidth, clipX1, clipX2), iclamp(centerX - halfWidth + 2 * halfWidth + 1, clipX1, clipX2), color);
959 }

1036 void IRAM_ATTR BitmappedDisplayController::fillPath(Path const & path, RGB888 const & color, Rect & updateRect)
1037 {
1038  const int clipX1 = paintState().absClippingRect.X1;
1039  const int clipY1 = paintState().absClippingRect.Y1;
1040  const int clipX2 = paintState().absClippingRect.X2;
1041  const int clipY2 = paintState().absClippingRect.Y2;
1042 
1043  const int origX = paintState().origin.X;
1044  const int origY = paintState().origin.Y;
1045 
1046  int minX = clipX1;
1047  int maxX = clipX2 + 1;
1048  int minY = INT_MAX;
1049  int maxY = 0;
1050  for (int i = 0; i < path.pointsCount; ++i) {
1051  int py = path.points[i].Y + origY;
1052  if (py < minY)
1053  minY = py;
1054  if (py > maxY)
1055  maxY = py;
1056  }
1057  minY = tmax(clipY1, minY);
1058  maxY = tmin(clipY2, maxY);
1059 
1060  updateRect = updateRect.merge(Rect(minX, minY, maxX, maxY));
1061  hideSprites(updateRect);
1062 
1063  int16_t nodeX[path.pointsCount];
1064 
1065  for (int pixelY = minY; pixelY <= maxY; ++pixelY) {
1066 
1067  int nodes = 0;
1068  int j = path.pointsCount - 1;
1069  for (int i = 0; i < path.pointsCount; ++i) {
1070  int piy = path.points[i].Y + origY;
1071  int pjy = path.points[j].Y + origY;
1072  if ((piy < pixelY && pjy >= pixelY) || (pjy < pixelY && piy >= pixelY)) {
1073  int pjx = path.points[j].X + origX;
1074  int pix = path.points[i].X + origX;
1075  int a = (pixelY - piy) * (pjx - pix);
1076  int b = (pjy - piy);
1077  nodeX[nodes++] = pix + a / b + (((a < 0) ^ (b > 0)) && (a % b));
1078  }
1079  j = i;
1080  }
1081 
1082  int i = 0;
1083  while (i < nodes - 1) {
1084  if (nodeX[i] > nodeX[i + 1]) {
1085  tswap(nodeX[i], nodeX[i + 1]);
1086  if (i)
1087  --i;
1088  } else
1089  ++i;

1091 
1092  for (int i = 0; i < nodes; i += 2) {
1093  if (nodeX[i] >= maxX)
1094  break;
1095  if (nodeX[i + 1] > minX) {
1096  if (nodeX[i] < minX)
1097  nodeX[i] = minX;
1098  if (nodeX[i + 1] > maxX)
1099  nodeX[i + 1] = maxX;
1100  rawFillRow(pixelY, nodeX[i], nodeX[i + 1] - 1, color);
1101  }
1102  }
1103  }
1104

1106  m_primDynMemPool.free((void*)path.points);
