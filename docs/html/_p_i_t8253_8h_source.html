34 class PIT8253 {
35 
36 public:
37 
38  typedef void (*ChangeOut)(void * context, int timerIndex);
39  typedef void (*Tick)(void * context, int ticks);
40 
41  struct TimerInfo {
42  bool BCD; // BCD mode
43  int32_t mode; // Timer mode
44  int32_t RLMode; // Read/Load mode
45  int32_t resetHolding; // Holding area for timer reset count
46  int32_t resetCount; // Reload value when count is zero
47  int32_t count; // Current timer counter
48  int32_t latch; // Latched timer count (-1 = not latched)
49  bool LSBToggle; // true: Read load LSB, false: Read load MSB
50  bool out; // out state
51  bool gate; // date (1 = timer running)
52  bool running; // counting down in course
53  };
54

58  void setCallbacks(void * context, ChangeOut changeOut, Tick tick) {

60  m_changeOut = changeOut;
61  m_tick = tick;
62  }

65 
66  void reset();

74  bool getGate(int timerIndex) { return m_timer[timerIndex].gate; }
75 
76  void setGate(int timerIndex, bool value);
