82 void VGADirectController::allocateViewPort()
83 {
84  m_lines = (uint8_t**) heap_caps_malloc(sizeof(uint8_t*) * m_linesCount, MALLOC_CAP_8BIT | MALLOC_CAP_INTERNAL);
85  m_lines[0] = (uint8_t*) heap_caps_malloc(m_viewPortWidth * m_linesCount, MALLOC_CAP_DMA);
86  for (int i = 1; i < m_linesCount; ++i)
87  m_lines[i] = m_lines[0] + i * m_viewPortWidth;
88 }

114 void VGADirectController::run()
115 {
116  // must be started before interrupt alloc
117  startGPIOStream();
118 
119  s_scanLine = 0;
120 
121  // ESP_INTR_FLAG_LEVEL1: should be less than PS2Controller interrupt level, necessary when running on the same core
122  if (m_isr_handle == nullptr) {
123  CoreUsage::setBusiestCore(FABGLIB_VIDEO_CPUINTENSIVE_TASKS_CORE);
124  esp_intr_alloc_pinnedToCore(ETS_I2S1_INTR_SOURCE, ESP_INTR_FLAG_LEVEL1 | ESP_INTR_FLAG_IRAM, ISRHandler, this, &m_isr_handle, FABGLIB_VIDEO_CPUINTENSIVE_TASKS_CORE);
125  I2S1.int_clr.val = 0xFFFFFFFF;
126  I2S1.int_ena.out_eof = 1;
127  }
128 }

131 void VGADirectController::onSetupDMABuffer(lldesc_t volatile * buffer, bool isStartOfVertFrontPorch, int scan, bool isVisible, int visibleRow)
132 {
133  if (isVisible) {
134  buffer->buf = (uint8_t *) m_lines[visibleRow % m_linesCount];
135 
136  // generate interrupt every half m_linesCount
137  if ((scan == 0 && (visibleRow % (m_linesCount / 2)) == 0)) {
138  if (visibleRow == 0)
139  s_frameResetDesc = buffer;
140  buffer->eof = 1;
141  }
142  }

246 void IRAM_ATTR VGADirectController::ISRHandler(void * arg)
247 {
248  #if FABGLIB_VGAXCONTROLLER_PERFORMANCE_CHECK
249  auto s1 = getCycleCount();
250  #endif
251 
252  if (I2S1.int_st.out_eof) {
253 
254  auto ctrl = (VGADirectController *) arg;
255 
256  auto viewPortHeight = ctrl->m_viewPortHeight;
257 
258  auto desc = (volatile lldesc_t*) I2S1.out_eof_des_addr;
259 
260  if (desc == s_frameResetDesc) {
261  s_scanLine = 0;
262  s_VSync = false;
263  }
264 
265  int linesCount = ctrl->m_linesCount;
266  int scanLine = (s_scanLine + linesCount / 2) % viewPortHeight;
267 
268  const auto lineIndex = scanLine & (linesCount - 1);
269 
270  ctrl->m_drawScanlineCallback(ctrl->m_drawScanlineArg, (uint8_t*)(ctrl->m_lines[lineIndex]), scanLine);
271 
272  s_scanLine += linesCount / 2;
273 
274  if (s_scanLine >= viewPortHeight)
275  s_VSync = true;
276 
277  }
278 
279  #if FABGLIB_VGAXCONTROLLER_PERFORMANCE_CHECK
280  s_vgapalctrlcycles += getCycleCount() - s1;
281  #endif
282 
283  I2S1.int_clr.val = I2S1.int_st.val;
284 }
