211 void Terminal::activate(TerminalTransition transition)
212 {
213  xSemaphoreTake(m_mutex, portMAX_DELAY);
214  if (s_activeTerminal != this) {
215 
216  if (s_activeTerminal && transition != TerminalTransition::None) {
217  if (m_bitmappedDisplayController) {
218  // bitmapped controller, use Canvas to perform the animation
219  s_activeTerminal = nullptr;
220  switch (transition) {
221  case TerminalTransition::LeftToRight:
222  for (int x = 0; x < m_columns; ++x) {
223  m_canvas->scroll(m_font.width, 0);
224  m_canvas->setOrigin(-m_font.width * (m_columns - x - 1), 0);
225  for (int y = 0; y < m_rows; ++y)
226  m_canvas->renderGlyphsBuffer(m_columns - x - 1, y, &m_glyphsBuffer);
227  m_canvas->waitCompletion(false);
228  vTaskDelay(2 / portTICK_PERIOD_MS);
229  }
230  break;
231  case TerminalTransition::RightToLeft:
232  for (int x = 0; x < m_columns; ++x) {
233  m_canvas->scroll(-m_font.width, 0);
234  m_canvas->setOrigin(m_font.width * (m_columns - x - 1), 0);
235  for (int y = 0; y < m_rows; ++y)
236  m_canvas->renderGlyphsBuffer(x, y, &m_glyphsBuffer);
237  m_canvas->waitCompletion(false);
238  vTaskDelay(2 / portTICK_PERIOD_MS);
239  }
240  break;
241  default:
242  break;
243  }
244  } else {
245  // textual controller, use temporary buffer to perform animation
246  auto txtCtrl = static_cast<TextualDisplayController*>(m_displayController);
247  auto map = (uint32_t*) heap_caps_malloc(sizeof(uint32_t) * m_columns * m_rows, MALLOC_CAP_8BIT | MALLOC_CAP_INTERNAL);
248  memcpy(map, s_activeTerminal->m_glyphsBuffer.map, sizeof(uint32_t) * m_columns * m_rows);
249  txtCtrl->enableCursor(false);
250  txtCtrl->setTextMap(map, m_rows);
251  switch (transition) {
252  case TerminalTransition::LeftToRight:
253  for (int x = 0; x < m_columns; ++x) {
254  for (int y = 0; y < m_rows; ++y) {
255  memmove(map + y * m_columns + 1, map + y * m_columns, sizeof(uint32_t) * (m_columns - 1));
256  map[y * m_columns] = m_glyphsBuffer.map[y * m_columns + m_columns - x - 1];
257  }
258  vTaskDelay(5);
259  }
260  break;
261  case TerminalTransition::RightToLeft:
262  for (int x = 0; x < m_columns; ++x) {
263  for (int y = 0; y < m_rows; ++y) {
264  memmove(map + y * m_columns, map + y * m_columns + 1, sizeof(uint32_t) * (m_columns - 1));
265  map[y * m_columns + m_columns - 1] = m_glyphsBuffer.map[y * m_columns + x];
266  }
267  vTaskDelay(5);
268  }
269  break;
270  default:
271  break;
272  }
273  txtCtrl->setTextMap(m_glyphsBuffer.map, m_rows);
274  free(map);
275  }
276  }
277 
278  s_activeTerminal = this;
279  vTaskResume(m_keyboardReaderTaskHandle);
280  syncDisplayController();
281  }
282  xSemaphoreGive(m_mutex);
283 }

497 // connect to UART2
498 void Terminal::connectSerialPort(uint32_t baud, uint32_t config, int rxPin, int txPin, FlowControl flowControl, bool inverted)
499 {
500  uart_dev_t * uart = (volatile uart_dev_t *) DR_REG_UART2_BASE;
501 
502  bool initialSetup = !m_uart;
503 
504  if (initialSetup) {
505  // uart not configured, configure now
506 
507  #ifdef ARDUINO
508  Serial2.end();
509  #endif
510 
511  m_uart = true;
512 
513  DPORT_SET_PERI_REG_MASK(DPORT_PERIP_CLK_EN_REG, DPORT_UART2_CLK_EN);
514  DPORT_CLEAR_PERI_REG_MASK(DPORT_PERIP_RST_EN_REG, DPORT_UART2_RST);
515 
516  // flush
517  uartFlushTXFIFO();
518  uartFlushRXFIFO();
519 
520  // TX/RX Pin direction
521  configureGPIO(int2gpio(rxPin), GPIO_MODE_INPUT);
522  configureGPIO(int2gpio(txPin), GPIO_MODE_OUTPUT);
523 
524  // RX interrupt
525  uart->conf1.rxfifo_full_thrhd = 1; // an interrupt for each character received
526  uart->conf1.rx_tout_thrhd = 2; // actually not used
527  uart->conf1.rx_tout_en = 0; // timeout not enabled
528  uart->int_ena.rxfifo_full = 1; // interrupt on FIFO full (1 character - see rxfifo_full_thrhd)
529  uart->int_ena.frm_err = 1; // interrupt on frame error
530  uart->int_ena.rxfifo_tout = 0; // no interrupt on rx timeout (see rx_tout_en and rx_tout_thrhd)
531  uart->int_ena.parity_err = 1; // interrupt on rx parity error
532  uart->int_ena.rxfifo_ovf = 1; // interrupt on rx overflow
533  uart->int_clr.val = 0xffffffff;
534  esp_intr_alloc(ETS_UART2_INTR_SOURCE, 0, uart_isr, this, nullptr);
535 
536  // setup FIFOs size
537  uart->mem_conf.rx_size = 3; // RX: 384 bytes (this is the max for UART2)
538  uart->mem_conf.tx_size = 1; // TX: 128 bytes
539 
540  if (!m_keyboardReaderTaskHandle && m_keyboard->isKeyboardAvailable())
541  xTaskCreate(&keyboardReaderTask, "", Terminal::keyboardReaderTaskStackSize, this, FABGLIB_KEYBOARD_READER_TASK_PRIORITY, &m_keyboardReaderTaskHandle);
542  }
543 
544  m_autoXONOFF = (flowControl == FlowControl::Software);
545 
546  // set baud rate
547  uint32_t clk_div = (getApbFrequency() << 4) / baud;
548  uart->clk_div.div_int = clk_div >> 4;
549  uart->clk_div.div_frag = clk_div & 0xf;
550 
551  // frame
552  uart->conf0.val = config;
553  if (uart->conf0.stop_bit_num == 0x3) {
554  uart->conf0.stop_bit_num = 1;
555  uart->rs485_conf.dl1_en = 1;
556  }
557 
558  // TX/RX Pin logic
559  gpio_matrix_in(rxPin, U2RXD_IN_IDX, inverted);
560  gpio_matrix_out(txPin, U2TXD_OUT_IDX, inverted, false);
561 
562  // Flow Control
563  uart->flow_conf.sw_flow_con_en = 0;
564  uart->flow_conf.xonoff_del = 0;
565  if (flowControl == FlowControl::Software) {
566  // we actually use manual software control, using send_xon/send_xoff bits to send control characters
567  // because we have to check both RX-FIFO and input queue
568  uart->swfc_conf.xon_threshold = 0;
569  uart->swfc_conf.xoff_threshold = 0;
570  uart->swfc_conf.xon_char = ASCII_XON;
571  uart->swfc_conf.xoff_char = ASCII_XOFF;
572  if (initialSetup) {
573  // send an XON right now
574  m_XOFF = true;
575  uart->flow_conf.send_xon = 1;
576  }
577  }
578 
579  // APB Change callback (TODO?)
580  //addApbChangeCallback(this, uart_on_apb_change);
581 }
