97 int SineWaveformGenerator::getSample() {
98  if (m_frequency == 0 || duration() == 0) {
99  if (m_lastSample > 0)
100  --m_lastSample;
101  else if (m_lastSample < 0)
102  ++m_lastSample;
103  else
104  m_phaseAcc = 0;
105  return m_lastSample;
106  }
107 
108  // get sample (-128...+127)
109  uint32_t index = m_phaseAcc >> 11;
110  double fmul = (double)(m_phaseAcc & 0x7ff) / 2048.0;
111  int sample = sinTable[index] + (sinTable[index + 1] - sinTable[index]) * fmul;
112 
113  // process volume
114  sample = sample * volume() / 127;
115 
116  m_lastSample = sample;
117 
118  m_phaseAcc = (m_phaseAcc + m_phaseInc) & 0x7ffff;
119 
120  decDuration();
121 
122  return sample;
123 }

359 int VICNoiseGenerator::getSample()
360 {
361  if (duration() == 0) {
362  return 0;
363  }
364 
365  const int reduc = CLK / 8 / sampleRate(); // resample to sampleRate() (ie 16000Hz)
366 
367  int sample = 0;
368 
369  for (int i = 0; i < reduc; ++i) {
370 
371  if (m_counter >= 127) {
372 
373  // reset counter
374  m_counter = m_frequency;
375 
376  if (m_LFSR & 1)
377  m_outSR = ((m_outSR << 1) | ~(m_outSR >> 7));
378 
379  m_LFSR <<= 1;
380  int bit3 = (m_LFSR >> 3) & 1;
381  int bit12 = (m_LFSR >> 12) & 1;
382  int bit14 = (m_LFSR >> 14) & 1;
383  int bit15 = (m_LFSR >> 15) & 1;
384  m_LFSR |= (bit3 ^ bit12) ^ (bit14 ^ bit15);
385  } else
386  ++m_counter;
387 
388  sample += m_outSR & 1 ? 127 : -128;
389  }
390 
391  // simple mean of all samples
392 
393  sample = sample / reduc;
394 
395  // process volume
396  sample = sample * volume() / 127;
397 
398  decDuration();
399 
400  return sample;
401 }

544 bool SoundGenerator::forcePlay(bool value)
545 {
546  bool isPlaying = actualPlaying();
547  if (value) {
548  // play
549  if (!isPlaying) {
550  if (!m_waveGenTaskHandle)
551  xTaskCreate(waveGenTask, "", WAVEGENTASK_STACK_SIZE, this, 5, &m_waveGenTaskHandle);
552  m_state = SoundGeneratorState::RequestToPlay;
553  xTaskNotifyGive(m_waveGenTaskHandle);
554  }
555  } else {
556  // stop
557  if (isPlaying) {
558  // request task to suspend itself when possible
559  m_state = SoundGeneratorState::RequestToStop;
560  // wait for task switch to suspend state (TODO: is there a better way?)
561  while (m_state != SoundGeneratorState::Stop)
562  vTaskDelay(1);
563  }
564  }
565  return isPlaying;
566 }

620 void IRAM_ATTR SoundGenerator::waveGenTask(void * arg)
621 {
622  SoundGenerator * soundGenerator = (SoundGenerator*) arg;
623 
624  i2s_set_clk(I2S_NUM_0, soundGenerator->m_sampleRate, I2S_BITS_PER_SAMPLE_16BIT, I2S_CHANNEL_MONO);
625 
626  uint16_t * buf = soundGenerator->m_sampleBuffer;
627 
628  // number of mute (without channels to play) cycles
629  int muteCyclesCount = 0;
630 
631  while (true) {
632 
633  // suspend?
634  if (soundGenerator->m_state == SoundGeneratorState::RequestToStop || soundGenerator->m_state == SoundGeneratorState::Stop) {
635  soundGenerator->m_state = SoundGeneratorState::Stop;
636  while (soundGenerator->m_state == SoundGeneratorState::Stop)
637  ulTaskNotifyTake(pdTRUE, portMAX_DELAY); // wait for "give"
638  }
639 
640  // mutize output?
641  if (soundGenerator->m_channels == nullptr && muteCyclesCount >= 8) {
642  soundGenerator->m_state = SoundGeneratorState::Stop;
643  while (soundGenerator->m_state == SoundGeneratorState::Stop)
644  ulTaskNotifyTake(pdTRUE, portMAX_DELAY); // wait for "give"
645  }
646 
647  soundGenerator->m_state = SoundGeneratorState::Playing;
648 
649  int mainVolume = soundGenerator->volume();
650 
651  for (int i = 0; i < FABGL_SAMPLE_BUFFER_SIZE; ++i) {
652  int sample = 0, tvol = 0;
653  for (auto g = soundGenerator->m_channels; g; ) {
654  if (g->enabled()) {
655  sample += g->getSample();
656  tvol += g->volume();
657  } else if (g->duration() == 0 && g->autoDetach()) {
658  auto curr = g;
659  g = g->next; // setup next item before detaching this one
660  soundGenerator->detachNoSuspend(curr);
661  continue; // bypass "g = g->next;"
662  }
663  g = g->next;
664  }
665 
666  int avol = tvol ? imin(127, 127 * 127 / tvol) : 127;
667  sample = sample * avol / 127;
668  sample = sample * mainVolume / 127;
669 
670  buf[i + (i & 1 ? -1 : 1)] = (127 + sample) << 8;
671  }
672 
673  size_t bytes_written;
674  i2s_write(I2S_NUM_0, buf, FABGL_SAMPLE_BUFFER_SIZE * sizeof(uint16_t), &bytes_written, portMAX_DELAY);
675 
676  muteCyclesCount = soundGenerator->m_channels == nullptr ? muteCyclesCount + 1 : 0;
677  }
678 }
