79 #define SSD1306_SETPIXEL(x, y) (m_screenBuffer[(x) + ((y) >> 3) * m_viewPortWidth] |= (1 << ((y) & 7)))
80 #define SSD1306_CLEARPIXEL(x, y) (m_screenBuffer[(x) + ((y) >> 3) * m_viewPortWidth] &= ~(1 << ((y) & 7)))

159 void SSD1306Controller::setResolution(char const * modeline, int viewPortWidth, int viewPortHeight, bool doubleBuffered)
160 {
161  char label[32];
162  int pos = 0, swidth, sheight;
163  int count = sscanf(modeline, "\"%[^\"]\" %d %d %n", label, &swidth, &sheight, &pos);
164  if (count != 3 || pos == 0)
165  return; // invalid modeline
166 
167  m_screenWidth = swidth;
168  m_screenHeight = sheight;
169  m_screenCol = 0;
170  m_screenRow = 0;
171 
172  // inform base class about screen size
173  setScreenSize(m_screenWidth, m_screenHeight);
174 
175  setDoubleBuffered(doubleBuffered);
176 
177  m_viewPortWidth = viewPortWidth < 0 ? m_screenWidth : viewPortWidth;
178  m_viewPortHeight = viewPortHeight < 0 ? m_screenHeight : viewPortHeight;
179 
180  resetPaintState();
181 
182  SSD1306_hardReset();
183 
184  if (!SSD1306_softReset())
185  return;
186 
187  allocScreenBuffer();
188 
189  // setup update task
190  xTaskCreate(&updateTaskFunc, "", SSD1306_UPDATETASK_STACK, this, SSD1306_UPDATETASK_PRIORITY, &m_updateTaskHandle);
191 
192  // allows updateTaskFunc() to run
193  m_updateTaskFuncSuspended = 0;
194 }

222 bool SSD1306Controller::SSD1306_sendData(uint8_t * buf, int count, uint8_t ctrl)
223 {
224  int bufSize = m_i2c->getMaxBufferLength();
225  uint8_t sbuf[bufSize];
226  sbuf[0] = ctrl;
227  while (count > 0) {
228  int bToSend = imin(bufSize - 1, count);
229  memcpy(&sbuf[1], buf, bToSend);
230  if (!m_i2c->write(m_i2cAddress, sbuf, bToSend + 1, SSD1306_I2C_FREQUENCY, SSD1306_I2C_TIMEOUT))
231  return false;
232  count -= bToSend;
233  buf += bToSend;
234  ets_delay_us(2); // delay 2uS (standing to SSD1306 doc should be 1.3uS, before new transmission can start)
235  }
236  return true;
237 }

274 // soft reset SSD1306
275 bool SSD1306Controller::SSD1306_softReset()
276 {
277  SSD1306_sendCmd(SSD1306_DISPLAYOFF);
278  SSD1306_sendCmd(SSD1306_SETDISPLAYCLOCKDIV, 0x80);
279  SSD1306_sendCmd(SSD1306_SETMULTIPLEX, m_screenHeight - 1);
280  SSD1306_sendCmd(SSD1306_SETDISPLAYOFFSET, 0);
281  SSD1306_sendCmd(SSD1306_SETSTARTLINE);
282  SSD1306_sendCmd(SSD1306_CHARGEPUMP, 0x14); // 0x14 = SWITCHCAPVCC, 0x10 = EXTERNALVCC
283  SSD1306_sendCmd(SSD1306_MEMORYMODE, 0b100); // 0b100 = page addressing mode
284  setupOrientation();
285  if (m_screenHeight == 64) {
286  SSD1306_sendCmd(SSD1306_SETCOMPINS, 0x12);
287  SSD1306_sendCmd(SSD1306_SETCONTRAST, 0xCF); // max: 0xCF = SWITCHCAPVCC, 0x9F = EXTERNALVCC
288  } else if (m_screenHeight == 32) {
289  SSD1306_sendCmd(SSD1306_SETCOMPINS, 0x02);
290  SSD1306_sendCmd(SSD1306_SETCONTRAST, 0x8F);
291  }
292  SSD1306_sendCmd(SSD1306_SETPRECHARGE, 0xF1); // 0xF1 = SWITCHCAPVCC, 0x22 = EXTERNALVCC
293  SSD1306_sendCmd(SSD1306_SETVCOMDETECT, 0x40);
294  SSD1306_sendCmd(SSD1306_DISPLAYALLON_RESUME);
295  SSD1306_sendCmd(SSD1306_NORMALDISPLAY);
296  SSD1306_sendCmd(SSD1306_DEACTIVATE_SCROLL);
297  return SSD1306_sendCmd(SSD1306_DISPLAYON);
298 }

301 void SSD1306Controller::setupOrientation()
302 {
303  switch (m_orientation) {
304  case SSD1306Orientation::Normal:
305  SSD1306_sendCmd(SSD1306_SEGREMAP | 0x1);
306  SSD1306_sendCmd(SSD1306_COMSCANDEC);
307  break;
308  case SSD1306Orientation::ReverseHorizontal:
309  SSD1306_sendCmd(SSD1306_SEGREMAP | 0x0);
310  SSD1306_sendCmd(SSD1306_COMSCANDEC);
311  break;
312  case SSD1306Orientation::ReverseVertical:
313  SSD1306_sendCmd(SSD1306_SEGREMAP | 0x1);
314  SSD1306_sendCmd(SSD1306_COMSCANINC);
315  break;
316  case SSD1306Orientation::Rotate180:
317  SSD1306_sendCmd(SSD1306_SEGREMAP | 0x0);
318  SSD1306_sendCmd(SSD1306_COMSCANINC);
319  break;
320  }
321 }

332 void SSD1306Controller::SSD1306_sendScreenBuffer(Rect updateRect)
333 {
334  // align visible screen row to page (steps of 8 rows)
335  const int screenRow = m_screenRow & ~7;
336 
337  // visible area
338  const Rect scrRect = Rect(m_screenCol, screenRow, m_screenCol + m_screenWidth - 1, screenRow + m_screenHeight - 1);
339 
340  // align rectangle to update to pages (0, 8, 16...)
341  updateRect.Y1 &= ~7;
342  updateRect.Y2 = (updateRect.Y2 + 7) & ~7;
343 
344  // does the visible area intersect with area to update?
345  if (scrRect.intersects(updateRect)) {
346 
347  // intersection between visible area and rectangle to update
348  Rect r = updateRect.intersection(scrRect);
349 
350  // horizontal screen update limits
351  const int screenX1 = r.X1 - m_screenCol;
352  const int screenX2 = r.X2 - m_screenCol;
353 
354  // send one page (8 rows) at the time
355  for (int y = r.Y1; y <= r.Y2; y += 8) {
356  int screenY = y - screenRow;
357  if (screenY >= 0) {
358  int page = screenY >> 3;
359  if (!(SSD1306_sendCmd(SSD1306_PAGEADDR, page, page) && SSD1306_sendCmd(SSD1306_COLUMNADDR, screenX1, screenX2)))
360  break; // address selection failed, try with next page
361  SSD1306_sendData(m_screenBuffer + r.X1 + (y >> 3) * m_viewPortWidth, r.width(), 0x40);
362  }
363  }
364 
365  }
366 }

382 void SSD1306Controller::updateTaskFunc(void * pvParameters)
383 {
384  SSD1306Controller * ctrl = (SSD1306Controller*) pvParameters;
385 
386  while (true) {
387 
388  ctrl->waitForPrimitives();
389 
390  // primitive processing blocked?
391  if (ctrl->m_updateTaskFuncSuspended > 0)
392  ulTaskNotifyTake(true, portMAX_DELAY); // yes, wait for a notify
393 
394  ctrl->m_updateTaskRunning = true;
395 
396  Rect updateRect = Rect(SHRT_MAX, SHRT_MAX, SHRT_MIN, SHRT_MIN);
397 
398  int64_t startTime = ctrl->backgroundPrimitiveTimeoutEnabled() ? esp_timer_get_time() : 0;
399  do {
400 
401  Primitive prim;
402  if (ctrl->getPrimitive(&prim) == false)
403  break;
404 
405  ctrl->execPrimitive(prim, updateRect, false);
406 
407  if (ctrl->m_updateTaskFuncSuspended > 0)
408  break;
409 
410  } while (!ctrl->backgroundPrimitiveTimeoutEnabled() || (startTime + SSD1306_BACKGROUND_PRIMITIVE_TIMEOUT > esp_timer_get_time()));
411 
412  ctrl->showSprites(updateRect);
413 
414  ctrl->m_updateTaskRunning = false;
415 
416  if (!ctrl->isDoubleBuffered())
417  ctrl->SSD1306_sendScreenBuffer(updateRect);
418  }
419 }
