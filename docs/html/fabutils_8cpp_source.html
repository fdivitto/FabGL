76 
77 // Integer square root by Halleck's method, with Legalize's speedup
78 int isqrt (int x)
79 {
80  if (x < 1)
81  return 0;
82  int squaredbit = 0x40000000;
83  int remainder = x;
84  int root = 0;
85  while (squaredbit > 0) {
86  if (remainder >= (squaredbit | root)) {
87  remainder -= (squaredbit | root);
88  root >>= 1;
89  root |= squaredbit;
90  } else
91  root >>= 1;
92  squaredbit >>= 2;
93  }
94  return root;
95 }

260 // Sutherland-Cohen line clipping algorithm
261 
262 static int clipLine_code(int x, int y, Rect const & clipRect)
263 {
264  int code = 0;
265  if (x < clipRect.X1)
266  code = 1;
267  else if (x > clipRect.X2)
268  code = 2;
269  if (y < clipRect.Y1)
270  code |= 4;
271  else if (y > clipRect.Y2)
272  code |= 8;
273  return code;
274 }
275 
276 // false = line is out of clipping rect
277 // true = line intersects or is inside the clipping rect (x1, y1, x2, y2 are changed if checkOnly=false)
278 bool clipLine(int & x1, int & y1, int & x2, int & y2, Rect const & clipRect, bool checkOnly)
279 {
280  int newX1 = x1;
281  int newY1 = y1;
282  int newX2 = x2;
283  int newY2 = y2;
284  int topLeftCode = clipLine_code(newX1, newY1, clipRect);
285  int bottomRightCode = clipLine_code(newX2, newY2, clipRect);
286  while (true) {
287  if ((topLeftCode == 0) && (bottomRightCode == 0)) {
288  if (!checkOnly) {
289  x1 = newX1;
290  y1 = newY1;
291  x2 = newX2;
292  y2 = newY2;
293  }
294  return true;
295  } else if (topLeftCode & bottomRightCode) {
296  break;
297  } else {
298  int x = 0, y = 0;
299  int ncode = topLeftCode != 0 ? topLeftCode : bottomRightCode;
300  if (ncode & 8) {
301  x = newX1 + (newX2 - newX1) * (clipRect.Y2 - newY1) / (newY2 - newY1);
302  y = clipRect.Y2;
303  } else if (ncode & 4) {
304  x = newX1 + (newX2 - newX1) * (clipRect.Y1 - newY1) / (newY2 - newY1);
305  y = clipRect.Y1;
306  } else if (ncode & 2) {
307  y = newY1 + (newY2 - newY1) * (clipRect.X2 - newX1) / (newX2 - newX1);
308  x = clipRect.X2;
309  } else if (ncode & 1) {
310  y = newY1 + (newY2 - newY1) * (clipRect.X1 - newX1) / (newX2 - newX1);
311  x = clipRect.X1;
312  }
313  if (ncode == topLeftCode) {
314  newX1 = x;
315  newY1 = y;
316  topLeftCode = clipLine_code(newX1, newY1, clipRect);
317  } else {
318  newX2 = x;
319  newY2 = y;
320  bottomRightCode = clipLine_code(newX2, newY2, clipRect);
321  }
322  }
323  }
324  return false;
325 }

329 // removeRectangle
330 // remove "rectToRemove" from "mainRect", pushing remaining rectangles to "rects" stack
331 
332 void removeRectangle(Stack<Rect> & rects, Rect const & mainRect, Rect const & rectToRemove)
333 {
334  if (!mainRect.intersects(rectToRemove) || rectToRemove.contains(mainRect))
335  return;
336 
337  // top rectangle
338  if (mainRect.Y1 < rectToRemove.Y1)
339  rects.push(Rect(mainRect.X1, mainRect.Y1, mainRect.X2, rectToRemove.Y1 - 1));
340 
341  // bottom rectangle
342  if (mainRect.Y2 > rectToRemove.Y2)
343  rects.push(Rect(mainRect.X1, rectToRemove.Y2 + 1, mainRect.X2, mainRect.Y2));
344 
345  // left rectangle
346  if (mainRect.X1 < rectToRemove.X1)
347  rects.push(Rect(mainRect.X1, tmax(rectToRemove.Y1, mainRect.Y1), rectToRemove.X1 - 1, tmin(rectToRemove.Y2, mainRect.Y2)));
348 
349  // right rectangle
350  if (mainRect.X2 > rectToRemove.X2)
351  rects.push(Rect(rectToRemove.X2 + 1, tmax(rectToRemove.Y1, mainRect.Y1), mainRect.X2, tmin(rectToRemove.Y2, mainRect.Y2)));
352 }

371 // rgb222_to_hsv
372 // R, G, B in the 0..3 range
373 void rgb222_to_hsv(int R, int G, int B, double * h, double * s, double * v)
374 {
375  double r = R / 3.0;
376  double g = G / 3.0;
377  double b = B / 3.0;
378  double cmax = tmax<double>(tmax<double>(r, g), b);
379  double cmin = tmin<double>(tmin<double>(r, g), b);
380  double diff = cmax - cmin;
381  if (cmax == cmin)
382  *h = 0;
383  else if (cmax == r)
384  *h = fmod((60.0 * ((g - b) / diff) + 360.0), 360.0);
385  else if (cmax == g)
386  *h = fmod((60.0 * ((b - r) / diff) + 120.0), 360.0);
387  else if (cmax == b)
388  *h = fmod((60.0 * ((r - g) / diff) + 240.0), 360.0);
389  *s = cmax == 0 ? 0 : (diff / cmax) * 100.0;
390  *v = cmax * 100.0;
391 }
