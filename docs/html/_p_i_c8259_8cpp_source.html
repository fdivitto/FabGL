31 #define ICW1_IC4 0x01 // 1 = ICW4 needed
32 #define ICW1_SNGL 0x02 // 1 = single mode, 0 = cascade mode
33 #define ICW1_LTIM 0x08 // 1 = level trig imput, 0 = edge trig input
34 
35 #define ICW4_PM 0x01 // 1 = x86, 0 = 8085
36 #define ICW4_AEOI 0x02 // 1 = auto EOI, 0 = normal EOI
37 #define ICW4_MS 0x04 // valid when ICW4_BUF = 1: 0 = buffered mode slave, 1 = buffered mode master
38 #define ICW4_BUF 0x08 // 1 = buffered mode, 0 = non buffered mode
39 #define ICW4_SFNM 0x10 // 1 = special fully nested mode
40 
41 #define OCW2_EOI 0x20 // 1 = EOI command
42 #define OCW2_SL 0x40 // 1 = with EOI means Specific, with R means Set Priority
43 #define OCW2_R 0x80 // 1 = with EOI means Rotate, with SL means Set Priority
44 
45 #define OCW3_RIS 0x01 // 1 = with OCW3_RR means Read IS register, 0 = with OCW3_RR means Read IR register
46 #define OCW3_RR 0x02 // 1 = OCW3_RIS specifies which register to read, 0 = no action
47 #define OCW3_POLL 0x04 // 1 = poll command
48 #define OCW3_SMM 0x08 // valid when OCW3_ESMM = 1: 0 = reset special mask, 1 = set special mask
49 
50 #define PORT0_ICW1 0x10 // 1 = presenting ICW1 on port 0, 0 = presenting OCW2 or OCW3 on port 0
51 #define PORT0_OCW3 0x08 // 1 = presenting OCW3 on port 0, 0 = presenting OCW2 on port 0

75 void PIC8259::write(int addr, uint8_t value)
76 {
77  switch (addr) {
78 
79  case 0:
80  if (value & PORT0_ICW1) {
81  // getting ICW1
82  //printf("ICW1 <= %02X\n", value);
83  //printf(" LTIM=%d ADI=%d SNGL=%d IC4=%d\n", (bool)(value & 0x08), (bool)(value & 0x04), (bool)(value & 0x02), (bool)(value & 0x01));
84  m_state = STATE_WAITING_ICW2;
85  m_state |= value & ICW1_IC4 ? STATE_WAITING_ICW4 : 0;
86  m_state |= value & ICW1_SNGL ? 0 : STATE_WAITING_ICW3;
87  } else if (value & PORT0_OCW3) {
88  // getting OCW3
89  //printf("OCW3 <= %02X\n", value);
90  //printf(" ESMM=%d SMM=%d P=%d RR=%d RIS=%d\n", (bool)(value & 0x40), (bool)(value & 0x20), (bool)(value & 0x04), (bool)(value & 0x02), (bool)(value & 0x01));
91  if (value & OCW3_RR)
92  m_readISR = value & OCW3_RIS;
93  } else {
94  // getting OCW2
95  //printf("OCW2 <= %02X\n", value);
96  //printf(" R=%d SL=%d EOI=%d L210=%d\n", (bool)(value & 0x80), (bool)(value & 0x40), (bool)(value & 0x20), value & 7);
97  if (value & OCW2_EOI) {
98  // perform EOI
99  performEOI();
100  }
101  }
102  break;
103 
104  case 1:
105  if (m_state & STATE_WAITING_ICW2) {
106  // getting ICW2
107  //printf("ICW2 <= %02X\n", value);
108  m_baseVector = value & 0xf8;
109  m_state &= ~STATE_WAITING_ICW2;
110  } else if (m_state & STATE_WAITING_ICW3) {
111  // getting ICW3
112  //printf("ICW3 <= %02X\n", value);
113  // not supported, nothing to do
114  m_state &= ~STATE_WAITING_ICW3;
115  } else if (m_state & STATE_WAITING_ICW4) {
116  // getting ICW4
117  //printf("ICW4 <= %02X\n", value);
118  //printf(" SFNM=%d BUF=%d M/S=%d AEOI=%d uPM=%d\n", (bool)(value & 0x10), (bool)(value & 0x08), (bool)(value & 0x04), (bool)(value & 0x02), (bool)(value & 0x01));
119  m_autoEOI = value & ICW4_AEOI;
120  m_state &= ~STATE_WAITING_ICW4;
121  } else {
122  // getting OCW1
123  //printf("OCW1 <= %02X\n", value);
124  m_IMR = value;
125  }
126  break;
127  }
128 }

151 // 256 = no bit set
152 int PIC8259::getHighestPriorityBitNum(uint8_t value)
153 {
154  if (value) {
155  for (int i = 0; i < 8; ++i)
156  if ((value >> i) & 1)
157  return i;
158  }
159  return 256;
160 }

163 // set as pending interrupt the higher priority bit in M_IRR
164 void PIC8259::setPendingInterrupt()
165 {
166  int highestInt = getHighestPriorityBitNum(m_IRR & ~m_IMR);
167  int servicingInt = getHighestPriorityBitNum(m_ISR);
168  //printf("PIC8259::setPendingInterrupt(), highestInt=%d, servicingInt=%d\n", highestInt, servicingInt);
169  if (highestInt < servicingInt) {
170  // interrupt in IRR has higher priority than the servicing one
171  m_pendingInterrupt = true;
172  m_pendingIR = highestInt;
173  }
174 }

177 // Device->8259: a device reports interrupt to 8259
178 bool PIC8259::signalInterrupt(int intnum)
179 {
180  //printf("PIC8259::signalInterrupt(%d)\n", intnum);
181  bool success = false;
182  intnum &= 7;
183  // already servicing?
184  if ((m_ISR & (1 << intnum)) == 0) {
185  // no, request interrupt
186  m_IRR |= 1 << intnum;
187  setPendingInterrupt();
188  success = true;
189  }
190  return success;
191 }

194 // CPU->8259: CPU acks the pending interrupt to 8259
195 void PIC8259::ackPendingInterrupt()
196 {
197  //printf("PIC8259::ackPendingInterrupt(), pending IR = %d\n", m_pendingIR);
198  int pendmsk = 1 << m_pendingIR;
199  m_ISR |= pendmsk; // set interrupt bit in ISR
200  m_IRR &= ~pendmsk; // reset interrupt bit in IRR
201  m_pendingInterrupt = false;
202  //printf(" ISR=%02X IRR=%02X\n", m_ISR, m_IRR);
203 }

206 // application reports EOI (of the higher interrupt)
207 void PIC8259::performEOI()
208 {
209  //printf("PIC8259::performEOI()\n");
210  // reset servicing int
211  int servicingInt = getHighestPriorityBitNum(m_ISR);
212  m_ISR &= ~(1 << servicingInt);
213  // is there another interrupt to ask for ack?
214  setPendingInterrupt();
215 }
