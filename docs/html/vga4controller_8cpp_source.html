54 static inline __attribute__((always_inline)) void VGA4_SETPIXELINROW(uint8_t * row, int x, int value) {
55  int brow = x >> 2;
56  int shift = 6 - (x & 3) * 2;
57  row[brow] ^= ((value << shift) ^ row[brow]) & (3 << shift);
58 }
59 
60 static inline __attribute__((always_inline)) int VGA4_GETPIXELINROW(uint8_t * row, int x) {
61  int brow = x >> 2;
62  int shift = 6 - (x & 3) * 2;
63  return (row[brow] >> shift) & 3;
64 }
65 
66 #define VGA4_INVERTPIXELINROW(row, x) (row)[(x) >> 2] ^= 3 << (6 - ((x) & 3) * 2)
67 
68 static inline __attribute__((always_inline)) void VGA4_SETPIXEL(int x, int y, int value) {
69  auto row = (uint8_t*) VGA4Controller::sgetScanline(y);
70  int brow = x >> 2;
71  int shift = 6 - (x & 3) * 2;
72  row[brow] ^= ((value << shift) ^ row[brow]) & (3 << shift);
73 }
74 
75 #define VGA4_GETPIXEL(x, y) VGA4_GETPIXELINROW((uint8_t*)VGA4Controller::s_viewPort[(y)], (x))
76 
77 #define VGA4_INVERT_PIXEL(x, y) VGA4_INVERTPIXELINROW((uint8_t*)VGA4Controller::s_viewPort[(y)], (x))

113 void VGA4Controller::setPaletteItem(int index, RGB888 const & color)
114 {
115  index %= 4;
116  m_palette[index] = color;
117  auto packed222 = RGB888toPackedRGB222(color);
118  for (int i = 0; i < 256; ++i) {
119  auto b = (uint8_t *) (m_packedPaletteIndexQuad_to_signals + i);
120  for (int j = 0; j < 4; ++j) {
121  auto aj = 6 - j * 2;
122  if (((i >> aj) & 3) == index) {
123  b[j ^ 2] = m_HVSync | packed222;
124  }
125  }
126  }

160 // parameters not checked
161 void VGA4Controller::rawFillRow(int y, int x1, int x2, uint8_t colorIndex)
162 {
163  uint8_t * row = (uint8_t*) m_viewPort[y];
164  // fill first pixels before full 8 bits word
165  int x = x1;
166  for (; x <= x2 && (x & 3) != 0; ++x) {
167  VGA4_SETPIXELINROW(row, x, colorIndex);
168  }
169  // fill whole 8 bits words (4 pixels)
170  if (x <= x2) {
171  int sz = (x2 & ~3) - x;
172  memset((void*)(row + x / 4), colorIndex | (colorIndex << 2) | (colorIndex << 4) | (colorIndex << 6), sz / 4);
173  x += sz;
174  }
175  // fill last unaligned pixels
176  for (; x <= x2; ++x) {
177  VGA4_SETPIXELINROW(row, x, colorIndex);
178  }
179 }

191 void VGA4Controller::rawCopyRow(int x1, int x2, int srcY, int dstY)
192 {
193  auto srcRow = (uint8_t*) m_viewPort[srcY];
194  auto dstRow = (uint8_t*) m_viewPort[dstY];
195  // copy first pixels before full 8 bits word
196  int x = x1;
197  for (; x <= x2 && (x & 3) != 0; ++x) {
198  VGA4_SETPIXELINROW(dstRow, x, VGA4_GETPIXELINROW(srcRow, x));
199  }
200  // copy whole 8 bits words (4 pixels)
201  auto src = (uint8_t*)(srcRow + x / 4);
202  auto dst = (uint8_t*)(dstRow + x / 4);
203  for (int right = (x2 & ~3); x < right; x += 4)
204  *dst++ = *src++;
205  // copy last unaligned pixels
206  for (x = (x2 & ~3); x <= x2; ++x) {
207  VGA4_SETPIXELINROW(dstRow, x, VGA4_GETPIXELINROW(srcRow, x));
208  }
209 }

212 void VGA4Controller::swapRows(int yA, int yB, int x1, int x2)
213 {
214  auto rowA = (uint8_t*) m_viewPort[yA];
215  auto rowB = (uint8_t*) m_viewPort[yB];
216  // swap first pixels before full 8 bits word
217  int x = x1;
218  for (; x <= x2 && (x & 3) != 0; ++x) {
219  uint8_t a = VGA4_GETPIXELINROW(rowA, x);
220  uint8_t b = VGA4_GETPIXELINROW(rowB, x);
221  VGA4_SETPIXELINROW(rowA, x, b);
222  VGA4_SETPIXELINROW(rowB, x, a);
223  }
224  // swap whole 8 bits words (4 pixels)
225  auto a = (uint8_t*)(rowA + x / 4);
226  auto b = (uint8_t*)(rowB + x / 4);
227  for (int right = (x2 & ~3); x < right; x += 4)
228  tswap(*a++, *b++);
229  // swap last unaligned pixels
230  for (x = (x2 & ~3); x <= x2; ++x) {
231  uint8_t a = VGA4_GETPIXELINROW(rowA, x);
232  uint8_t b = VGA4_GETPIXELINROW(rowB, x);
233  VGA4_SETPIXELINROW(rowA, x, b);
234  VGA4_SETPIXELINROW(rowB, x, a);
235  }
236 }

270 void VGA4Controller::HScroll(int scroll, Rect & updateRect)
271 {
272  hideSprites(updateRect);
273  uint8_t back = RGB888toPaletteIndex(getActualBrushColor());
274  uint8_t back4 = back | (back << 2) | (back << 4) | (back << 6);
275 
276  int Y1 = paintState().scrollingRegion.Y1;
277  int Y2 = paintState().scrollingRegion.Y2;
278  int X1 = paintState().scrollingRegion.X1;
279  int X2 = paintState().scrollingRegion.X2;
280 
281  int width = X2 - X1 + 1;
282  bool HScrolllingRegionAligned = ((X1 & 3) == 0 && (width & 3) == 0); // 4 pixels aligned
283 
284  if (scroll < 0) {
285  // scroll left
286  for (int y = Y1; y <= Y2; ++y) {
287  if (HScrolllingRegionAligned) {
288  // aligned horizontal scrolling region, fast version
289  uint8_t * row = (uint8_t*) (m_viewPort[y]) + X1 / 4;
290  for (int s = -scroll; s > 0;) {
291  if (s < 4) {
292  // scroll left by 1..3
293  int sz = width / 4;
294  uint8_t prev = back4;
295  for (int i = sz - 1; i >= 0; --i) {
296  uint8_t lowbits = prev >> (8 - s * 2);
297  prev = row[i];
298  row[i] = (row[i] << (s * 2)) | lowbits;
299  }
300  s = 0;
301  } else {
302  // scroll left by multiplies of 4
303  auto sc = s & ~3;
304  auto sz = width & ~3;
305  memmove(row, row + sc / 4, (sz - sc) / 4);
306  rawFillRow(y, X2 - sc + 1, X2, back);
307  s -= sc;
308  }
309  }
310  } else {
311  // unaligned horizontal scrolling region, fallback to slow version
312  auto row = (uint8_t*) m_viewPort[y];
313  for (int x = X1; x <= X2 + scroll; ++x)
314  VGA4_SETPIXELINROW(row, x, VGA4_GETPIXELINROW(row, x - scroll));
315  // fill right area with brush color
316  rawFillRow(y, X2 + 1 + scroll, X2, back);
317  }
318  }
319  } else if (scroll > 0) {
320  // scroll right
321  for (int y = Y1; y <= Y2; ++y) {
322  if (HScrolllingRegionAligned) {
323  // aligned horizontal scrolling region, fast version
324  uint8_t * row = (uint8_t*) (m_viewPort[y]) + X1 / 4;
325  for (int s = scroll; s > 0;) {
326  if (s < 4) {
327  // scroll right by 1..3
328  int sz = width / 4;
329  uint8_t prev = back4;
330  for (int i = 0; i < sz; ++i) {
331  uint8_t highbits = prev << (8 - s * 2);
332  prev = row[i];
333  row[i] = (row[i] >> (s * 2)) | highbits;
334  }
335  s = 0;
336  } else {
337  // scroll right by multiplies of 4
338  auto sc = s & ~3;
339  auto sz = width & ~3;
340  memmove(row + sc / 4, row, (sz - sc) / 4);
341  rawFillRow(y, X1, X1 + sc - 1, back);
342  s -= sc;
343  }
344  }
345  } else {
346  // unaligned horizontal scrolling region, fallback to slow version
347  auto row = (uint8_t*) m_viewPort[y];
348  for (int x = X2 - scroll; x >= X1; --x)
349  VGA4_SETPIXELINROW(row, x + scroll, VGA4_GETPIXELINROW(row, x));
350  // fill left area with brush color
351  rawFillRow(y, X1, X1 + scroll - 1, back);
352  }
353  }
354 
355  }
356 }

453 void IRAM_ATTR VGA4Controller::ISRHandler(void * arg)
454 {
455  #if FABGLIB_VGAXCONTROLLER_PERFORMANCE_CHECK
456  auto s1 = getCycleCount();
457  #endif
458 
459  auto ctrl = (VGA4Controller *) arg;
460 
461  if (I2S1.int_st.out_eof) {
462 
463  auto const desc = (lldesc_t*) I2S1.out_eof_des_addr;
464 
465  if (desc == s_frameResetDesc)
466  s_scanLine = 0;
467 
468  auto const width = ctrl->m_viewPortWidth;
469  auto const height = ctrl->m_viewPortHeight;
470  auto const packedPaletteIndexQuad_to_signals = (uint32_t const *) ctrl->m_packedPaletteIndexQuad_to_signals;
471  auto const lines = ctrl->m_lines;
472 
473  int scanLine = (s_scanLine + VGA4_LinesCount / 2) % height;
474 
475  auto lineIndex = scanLine & (VGA4_LinesCount - 1);
476 
477  for (int i = 0; i < VGA4_LinesCount / 2; ++i) {
478 
479  auto src = (uint8_t const *) s_viewPortVisible[scanLine];
480  auto dest = (uint32_t*) lines[lineIndex];
481 
482  // optimizazion warn: horizontal resolution must be a multiple of 16!
483  for (int col = 0; col < width; col += 16) {
484 
485  auto src1 = *(src + 0);
486  auto src2 = *(src + 1);
487  auto src3 = *(src + 2);
488  auto src4 = *(src + 3);
489 
490  PSRAM_HACK;
491 
492  auto v1 = packedPaletteIndexQuad_to_signals[src1];
493  auto v2 = packedPaletteIndexQuad_to_signals[src2];
494  auto v3 = packedPaletteIndexQuad_to_signals[src3];
495  auto v4 = packedPaletteIndexQuad_to_signals[src4];
496 
497  *(dest + 0) = v1;
498  *(dest + 1) = v2;
499  *(dest + 2) = v3;
500  *(dest + 3) = v4;
501 
502  dest += 4;
503  src += 4;
504  }
505 
506  ++lineIndex;
507  ++scanLine;
508  }
509 
510  s_scanLine += VGA4_LinesCount / 2;
511 
512  if (scanLine >= height && !ctrl->m_primitiveProcessingSuspended && spi_flash_cache_enabled() && ctrl->m_primitiveExecTask) {
513  // vertical sync, unlock primitive execution task
514  // warn: don't use vTaskSuspendAll() in primitive drawing, otherwise vTaskNotifyGiveFromISR may be blocked and screen will flick!
515  vTaskNotifyGiveFromISR(ctrl->m_primitiveExecTask, NULL);
516  }
517 
518  }
519 
520  #if FABGLIB_VGAXCONTROLLER_PERFORMANCE_CHECK
521  s_vgapalctrlcycles += getCycleCount() - s1;
522  #endif
523 
524  I2S1.int_clr.val = I2S1.int_st.val;
525 }
